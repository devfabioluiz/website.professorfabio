Módulo 5: Procedures e Funções Armazenados em PL/SQL

Tópicos do nosso plano de curso abordados nessa aula: Procedures armazenadas (CREATE OR REPLACE PROCEDURE, parâmetros IN/OUT/IN OUT), Funções (CREATE OR REPLACE FUNCTION, RETURN), Gerenciamento de transações no Oracle (COMMIT, ROLLBACK, SAVEPOINT), Pacotes (CREATE PACKAGE, CREATE PACKAGE BODY).

Procedures Armazenadas

Procedures armazenadas (Stored Procedures) são blocos de código PL/SQL nomeados e armazenados no banco de dados Oracle. Diferente de blocos anônimos, procedures são compiladas e armazenadas permanentemente, podendo ser chamadas diversas vezes com parâmetros diferentes.

Benefícios das Procedures

  • Reusabilidade: escreva uma vez, execute quantas vezes quiser com parâmetros diferentes
  • Performance: compiladas uma única vez e executadas em tempo de execução otimizado
  • Segurança: podem ser executadas com privilégios do criador (definer's rights) ou do chamador (invoker's rights)
  • Encapsulamento: lógica de negócio fica centralizada no banco, facilitando manutenção
  • Redução de tráfego: uma única chamada substitui múltiplos comandos SQL enviados pelo cliente
Importante: No Oracle, procedures NÃO retornam valor diretamente como funções. Elas podem retornar valores através de parâmetros OUT. Para retornar um valor, use funções (FUNCTION).

Criando Procedures — Sintaxe e Exemplos

A sintaxe básica de uma procedure no Oracle PL/SQL:


CREATE [OR REPLACE] PROCEDURE nome_procedure
    (parametro1 [IN | OUT | IN OUT] tipo_dado,
     parametro2 [IN | OUT | IN OUT] tipo_dado)
IS | AS
    -- seção declarativa (variáveis, cursores, exceções)
BEGIN
    -- seção executável (código PL/SQL)
    COMMIT;
EXCEPTION
    -- seção de tratamento de exceções (opcional)
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE;
END nome_procedure;
/

Vamos começar com uma procedure simples que atualiza o salário de um funcionário: 


-- Tabela de exemplo
CREATE TABLE funcionarios (
    id        NUMBER PRIMARY KEY,
    nome      VARCHAR2(100) NOT NULL,
    salario   NUMBER(10,2) NOT NULL CHECK (salario > 0),
    cargo     VARCHAR2(50)
);

-- Procedure simples: atualizar salário com percentual
CREATE OR REPLACE PROCEDURE atualizar_salario(
    p_id          NUMBER,
    p_percentual  NUMBER
)
IS
BEGIN
    UPDATE funcionarios
    SET salario = salario * (1 + p_percentual / 100)
    WHERE id = p_id;

    COMMIT;

    DBMS_OUTPUT.PUT_LINE('Salário do funcionário ' || p_id || ' atualizado em ' || p_percentual || '%');
END atualizar_salario;
/

Executando Procedures

Existem três formas de executar uma procedure no Oracle:


-- Método 1: Comando EXEC (SQL*Plus / SQL Developer)
EXEC atualizar_salario(1, 10);

-- Método 2: Comando EXECUTE
EXECUTE atualizar_salario(1, 10);

-- Método 3: Dentro de um bloco PL/SQL anônimo
BEGIN
    atualizar_salario(1, 10);
END;
/

-- Método 4: Dentro de outra procedure/function
CREATE OR REPLACE PROCEDURE reajuste_geral(p_percentual NUMBER)
IS
BEGIN
    FOR r IN (SELECT id FROM funcionarios) LOOP
        atualizar_salario(r.id, p_percentual);
    END LOOP;
END reajuste_geral;
/

Parâmetros IN, OUT e IN OUT

O Oracle PL/SQL suporta três modos de parâmetros em procedures e funções:

Resumo dos Modos:
  • IN (padrão): valor é passado para a procedure, não pode ser modificado dentro dela
  • OUT: procedure retorna um valor através do parâmetro, valor inicial é ignorado
  • IN OUT: valor é passado e pode ser modificado, retornando o novo valor

Parâmetros IN (padrão)

O modo padrão. O valor é passado para a procedure e é tratado como somente leitura: 


CREATE OR REPLACE PROCEDURE inserir_funcionario(
    p_nome    IN VARCHAR2,
    p_salario IN NUMBER,
    p_cargo   IN VARCHAR2
)
IS
BEGIN
    INSERT INTO funcionarios (id, nome, salario, cargo)
    VALUES (seq_funcionarios.NEXTVAL, p_nome, p_salario, p_cargo);

    COMMIT;

    DBMS_OUTPUT.PUT_LINE('Funcionário ' || p_nome || ' inserido com sucesso.');
END inserir_funcionario;
/

-- Chamando
EXEC inserir_funcionario('João Silva', 5500.00, 'Analista');

Parâmetros OUT

Permitem que a procedure retorne valores para o chamador:


CREATE OR REPLACE PROCEDURE obter_dados_funcionario(
    p_id     IN  NUMBER,
    p_nome   OUT VARCHAR2,
    p_salario OUT NUMBER
)
IS
BEGIN
    SELECT nome, salario
    INTO p_nome, p_salario
    FROM funcionarios
    WHERE id = p_id;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        p_nome := NULL;
        p_salario := NULL;
        DBMS_OUTPUT.PUT_LINE('Funcionário ' || p_id || ' não encontrado.');
END obter_dados_funcionario;
/

-- Para chamar com parâmetros OUT, é necessário um bloco PL/SQL
DECLARE
    v_nome   VARCHAR2(100);
    v_salario NUMBER;
BEGIN
    obter_dados_funcionario(1, v_nome, v_salario);
    DBMS_OUTPUT.PUT_LINE('Nome: ' || v_nome || ' | Salário: R$ ' || TO_CHAR(v_salario, 'FM999G990D00'));
END;
/

Parâmetros IN OUT

Recebem um valor, processam e retornam o resultado modificado:


CREATE OR REPLACE PROCEDURE formatar_nome(
    p_nome IN OUT VARCHAR2
)
IS
BEGIN
    -- Converte para maiúsculo e remove espaços extras
    p_nome := TRIM(UPPER(p_nome));
END formatar_nome;
/

-- Chamando com IN OUT
DECLARE
    v_nome VARCHAR2(100) := '  joão da silva  ';
BEGIN
    DBMS_OUTPUT.PUT_LINE('Antes: [' || v_nome || ']');
    formatar_nome(v_nome);
    DBMS_OUTPUT.PUT_LINE('Depois: [' || v_nome || ']');
END;
/
-- Resultado:
-- Antes: [  joão da silva  ]
-- Depois: [JOÃO DA SILVA]

Comparação entre IN, OUT e IN OUT


-- Procedure que demonstra os três modos simultaneamente
CREATE OR REPLACE PROCEDURE demo_parametros(
    p_entrada      IN     NUMBER,   -- Recebe valor (somente leitura)
    p_resultado    OUT    NUMBER,   -- Retorna valor
    p_acumulador   IN OUT NUMBER    -- Recebe e retorna valor
)
IS
BEGIN
    -- IN: pode ser lido mas não modificado
    p_resultado := p_entrada * 2;

    -- IN OUT: pode ser lido e modificado
    p_acumulador := p_acumulador + p_entrada;

    DBMS_OUTPUT.PUT_LINE('Entrada dobrada: ' || p_resultado);
    DBMS_OUTPUT.PUT_LINE('Acumulador atualizado: ' || p_acumulador);
END demo_parametros;
/

-- Testando
DECLARE
    v_saida      NUMBER;
    v_acumulador NUMBER := 100;
BEGIN
    demo_parametros(5, v_saida, v_acumulador);
    DBMS_OUTPUT.PUT_LINE('Resultado retornado: ' || v_saida);
    DBMS_OUTPUT.PUT_LINE('Acumulador final: ' || v_acumulador);
END;
/

Funções (Functions) — CREATE OR REPLACE FUNCTION

Funções são subprogramas que sempre retornam um valor. Diferente de procedures, funções podem ser usadas diretamente em comandos SQL (SELECT, WHERE, etc.).

Diferenças entre Procedure e Function

Característica Procedure Function
Retorno de valor Opcional (via OUT) Obrigatório (RETURN)
Uso em SQL Não permitido Permitido (com restrições)
Chamada EXEC / bloco PL/SQL Expressão / SELECT / bloco PL/SQL
Transações (COMMIT/ROLLBACK) Permitido Não permitido (em funções chamadas por SQL)
Parâmetros IN, OUT, IN OUT Apenas IN (convenção)

Sintaxe de Function


CREATE [OR REPLACE] FUNCTION nome_funcao
    (parametro1 tipo_dado, parametro2 tipo_dado)
RETURN tipo_retorno
IS
    -- seção declarativa
BEGIN
    -- seção executável
    RETURN valor;
EXCEPTION
    -- tratamento de erros (opcional)
END nome_funcao;
/

Exemplo: Função para Calcular Imposto


CREATE OR REPLACE FUNCTION calcular_imposto(
    p_salario NUMBER
)
RETURN NUMBER
IS
    v_imposto NUMBER := 0;
BEGIN
    -- Faixas progressivas
    IF p_salario <= 2259.20 THEN
        v_imposto := 0;
    ELSIF p_salario <= 2826.65 THEN
        v_imposto := p_salario * 0.075;
    ELSIF p_salario <= 3751.05 THEN
        v_imposto := p_salario * 0.15;
    ELSIF p_salario <= 4664.68 THEN
        v_imposto := p_salario * 0.225;
    ELSE
        v_imposto := p_salario * 0.275;
    END IF;

    RETURN ROUND(v_imposto, 2);
END calcular_imposto;
/

-- Testando a função
SELECT calcular_imposto(5000) AS imposto FROM DUAL;
-- Resultado: 1375

-- Usando diretamente em consultas SQL
SELECT nome, salario, calcular_imposto(salario) AS imposto_devido
FROM funcionarios
ORDER BY salario DESC;

Funções Determinísticas

Uma função é determinística quando sempre retorna o mesmo resultado para os mesmos parâmetros de entrada. Marcar uma função como DETERMINISTIC permite que o Oracle a utilize em function-based indexes e otimize chamadas repetidas. 


CREATE OR REPLACE FUNCTION bonus_por_cargo(
    p_cargo VARCHAR2
)
RETURN NUMBER
DETERMINISTIC
IS
BEGIN
    CASE p_cargo
        WHEN 'Gerente'   THEN RETURN 0.20;
        WHEN 'Analista'  THEN RETURN 0.10;
        WHEN 'Assistente' THEN RETURN 0.05;
        ELSE RETURN 0.02;
    END CASE;
END bonus_por_cargo;
/

-- Function-based index usando função determinística
CREATE INDEX idx_bonus_cargo ON funcionarios(bonus_por_cargo(cargo));

-- Exemplo: function que retorna tempo de empresa
CREATE OR REPLACE FUNCTION tempo_empresa_anos(
    p_data_admissao DATE
)
RETURN NUMBER
DETERMINISTIC
IS
BEGIN
    RETURN TRUNC(MONTHS_BETWEEN(SYSDATE, p_data_admissao) / 12, 2);
END tempo_empresa_anos;
/

Função com Retorno Complexo (Retornando uma Tabela)


-- Criando um tipo registro e uma coleção (table type)
CREATE TYPE funcionario_record AS OBJECT (
    nome    VARCHAR2(100),
    salario NUMBER(10,2),
    cargo   VARCHAR2(50)
);
/

CREATE TYPE funcionario_table AS TABLE OF funcionario_record;
/

CREATE OR REPLACE FUNCTION listar_por_cargo(
    p_cargo VARCHAR2
)
RETURN funcionario_table
PIPELINED
IS
BEGIN
    FOR r IN (
        SELECT nome, salario, cargo
        FROM funcionarios
        WHERE UPPER(cargo) = UPPER(p_cargo)
        ORDER BY salario DESC
    ) LOOP
        PIPE ROW (funcionario_record(r.nome, r.salario, r.cargo));
    END LOOP;

    RETURN;
END listar_por_cargo;
/

-- Consultando a função
SELECT * FROM TABLE(listar_por_cargo('Analista'));

Gerenciamento de Transações no Oracle

Transações são unidades lógicas de trabalho que contêm uma ou mais operações SQL. O Oracle gerencia transações automaticamente, mas o desenvolvedor pode controlá-las explicitamente com os comandos TCL (Transaction Control Language).

COMMIT

Torna permanentes todas as alterações feitas desde o último COMMIT ou desde o início da sessão: 


-- Demonstração de COMMIT
BEGIN
    INSERT INTO funcionarios (id, nome, salario, cargo)
    VALUES (100, 'Teste Commit', 3000, 'Analista');

    UPDATE funcionarios SET salario = salario * 1.1 WHERE cargo = 'Analista';

    -- As alterações agora são permanentes
    COMMIT;

    DBMS_OUTPUT.PUT_LINE('Transação confirmada com sucesso.');
END;
/

ROLLBACK

Desfaz todas as alterações pendentes (não commitadas):


-- Demonstração de ROLLBACK
BEGIN
    INSERT INTO funcionarios (id, nome, salario, cargo)
    VALUES (200, 'Teste Rollback', 3000, 'Assistente');

    UPDATE funcionarios SET salario = 99999 WHERE id = 1;

    DBMS_OUTPUT.PUT_LINE('Operações realizadas... mas serão desfeitas.');

    -- Desfaz tudo desde o último COMMIT
    ROLLBACK;

    DBMS_OUTPUT.PUT_LINE('Rollback executado. Nada foi salvo.');
END;
/

SAVEPOINT — Pontos de Salvamento Parciais

SAVEPOINT permite marcar pontos intermediários em uma transação, possibilitando rollback parcial: 


-- Demonstração prática de SAVEPOINT
BEGIN
    -- Insere um novo departamento
    INSERT INTO departamentos (id, nome) VALUES (10, 'P&D');

    SAVEPOINT depois_depto;

    -- Insere funcionários no novo departamento
    INSERT INTO funcionarios (id, nome, salario, cargo, depto_id)
    VALUES (300, 'Ana Pesquisadora', 7000, 'Pesquisador', 10);

    INSERT INTO funcionarios (id, nome, salario, cargo, depto_id)
    VALUES (301, 'Bruno Pesquisador', 6500, 'Pesquisador', 10);

    SAVEPOINT depois_funcionarios;

    -- Atualiza salários (ops, erro nos valores!)
    UPDATE funcionarios SET salario = salario * 0.5 WHERE depto_id = 10;

    -- Rollback parcial: desfaz apenas o UPDATE incorreto
    ROLLBACK TO depois_funcionarios;

    -- Confirma o resto (departamento criado e funcionários inseridos)
    COMMIT;

    DBMS_OUTPUT.PUT_LINE('Transação concluída — departamento e funcionários salvos, update desfeito.');
END;
/

Transferência entre Contas com SAVEPOINT + Validação


CREATE OR REPLACE PROCEDURE transferir_valor(
    p_conta_origem  NUMBER,
    p_conta_destino NUMBER,
    p_valor         NUMBER
)
IS
    v_saldo NUMBER;
BEGIN
    -- Verifica saldo da conta origem
    SELECT saldo INTO v_saldo FROM contas WHERE id = p_conta_origem;

    IF v_saldo < p_valor THEN
        RAISE_APPLICATION_ERROR(-20001, 'Saldo insuficiente na conta origem.');
    END IF;

    SAVEPOINT antes_debito;

    -- Debita da origem
    UPDATE contas SET saldo = saldo - p_valor WHERE id = p_conta_origem;

    -- Credita no destino
    UPDATE contas SET saldo = saldo + p_valor WHERE id = p_conta_destino;

    -- Se o destino ficou negativo (por alguma regra), desfaz
    SELECT saldo INTO v_saldo FROM contas WHERE id = p_conta_destino;

    IF v_saldo < 0 THEN
        ROLLBACK TO antes_debito;
        RAISE_APPLICATION_ERROR(-20002, 'Erro: destino não pode ficar negativo.');
    END IF;

    COMMIT;

    DBMS_OUTPUT.PUT_LINE('Transferência de R$ ' || p_valor || ' realizada com sucesso.');
END transferir_valor;
/

Transações Autônomas (PRAGMA AUTONOMOUS_TRANSACTION)

Transações autônomas são transações independentes que executam dentro de outra transação. Úteis para logging e auditoria — registram o que aconteceu mesmo que a transação principal sofra ROLLBACK. 


-- Tabela de log para auditoria
CREATE TABLE log_auditoria (
    id          NUMBER GENERATED ALWAYS AS IDENTITY,
    operacao    VARCHAR2(50),
    tabela      VARCHAR2(50),
    data_hora   TIMESTAMP DEFAULT SYSTIMESTAMP,
    usuario     VARCHAR2(50) DEFAULT USER,
    descricao   VARCHAR2(4000)
);

-- Procedure de log com transação autônoma
CREATE OR REPLACE PROCEDURE registrar_log(
    p_operacao  VARCHAR2,
    p_tabela    VARCHAR2,
    p_descricao VARCHAR2
)
IS
    PRAGMA AUTONOMOUS_TRANSACTION;
BEGIN
    INSERT INTO log_auditoria (operacao, tabela, descricao)
    VALUES (p_operacao, p_tabela, p_descricao);

    -- Commit apenas da transação autônoma (log)
    COMMIT;
END registrar_log;
/

-- Aplicando em uma procedure principal
CREATE OR REPLACE PROCEDURE excluir_funcionario(
    p_id NUMBER
)
IS
    v_nome VARCHAR2(100);
BEGIN
    SELECT nome INTO v_nome FROM funcionarios WHERE id = p_id;

    DELETE FROM funcionarios WHERE id = p_id;

    -- Registra no log (será salvo mesmo se a principal sofrer ROLLBACK)
    registrar_log('DELETE', 'FUNCIONARIOS', 'Funcionário excluído: ' || v_nome || ' (ID=' || p_id || ')');

    -- Se algo der errado...
    -- ROLLBACK; -- afeta apenas a transação principal, o log permanece!

    COMMIT;
END excluir_funcionario;
/

Pacotes (Packages) — Organização de Código PL/SQL

Pacotes são coleções de procedures, funções, variáveis, constantes, cursores e tipos relacionados logicamente. Um pacote é composto por duas partes: especificação (interface pública) e corpo (implementação).

Benefícios dos Pacotes

  • Encapsulamento: separa interface pública da implementação privada
  • Persistência de estado: variáveis do pacote mantêm seu valor durante a sessão
  • Sobrecarga (Overloading): múltiplas procedures/functions com mesmo nome e parâmetros diferentes
  • Performance: todo o pacote é carregado na memória na primeira chamada
  • Organização: agrupa código relacionado, facilitando manutenção

Criando um Pacote — Especificação (Package Specification)

A especificação contém as assinaturas (declarações) públicas:


CREATE OR REPLACE PACKAGE pkg_rh AS

    -- Constante do pacote
    c_salario_minimo CONSTANT NUMBER := 1518.00;

    -- Tipo personalizado
    TYPE t_funcionario IS RECORD (
        id      NUMBER,
        nome    VARCHAR2(100),
        salario NUMBER,
        cargo   VARCHAR2(50)
    );

    -- Procedure 1: Contratar funcionário
    PROCEDURE contratar(
        p_nome    VARCHAR2,
        p_salario NUMBER,
        p_cargo   VARCHAR2,
        p_depto_id NUMBER DEFAULT NULL
    );

    -- Procedure 2: Demitir funcionário
    PROCEDURE demitir(
        p_id     NUMBER,
        p_motivo VARCHAR2 DEFAULT 'Demissão sem justa causa'
    );

    -- Procedure 3: Promover funcionário
    PROCEDURE promover(
        p_id           NUMBER,
        p_novo_cargo   VARCHAR2,
        p_aumento_pct  NUMBER
    );

    -- Function 1: Calcular férias proporcionais
    FUNCTION calcular_ferias(
        p_salario        NUMBER,
        p_meses_trabalho NUMBER
    ) RETURN NUMBER;

    -- Function 2: Verificar se salário é compatível com o cargo
    FUNCTION validar_salario_cargo(
        p_cargo   VARCHAR2,
        p_salario NUMBER
    ) RETURN BOOLEAN;

    -- Function 3: Total de funcionários por departamento
    FUNCTION total_por_depto(
        p_depto_id NUMBER
    ) RETURN NUMBER;

END pkg_rh;
/

Criando o Corpo do Pacote (Package Body)

O corpo contém a implementação de todas as sub-rotinas declaradas na especificação, além de membros privados: 


CREATE OR REPLACE PACKAGE BODY pkg_rh AS

    -- ============ VARIÁVEL PRIVADA DO PACOTE ============
    v_ultima_contratacao VARCHAR2(100);

    -- ============ IMPLEMENTAÇÃO DAS PROCEDURES ============

    PROCEDURE contratar(
        p_nome    VARCHAR2,
        p_salario NUMBER,
        p_cargo   VARCHAR2,
        p_depto_id NUMBER DEFAULT NULL
    )
    IS
        v_id NUMBER;
    BEGIN
        -- Validações
        IF p_salario < c_salario_minimo THEN
            RAISE_APPLICATION_ERROR(-20010,
                'Salário de ' || TO_CHAR(p_salario, 'FM999G999D00') ||
                ' é menor que o salário mínimo de R$ ' ||
                TO_CHAR(c_salario_minimo, 'FM999G999D00'));
        END IF;

        IF NOT validar_salario_cargo(p_cargo, p_salario) THEN
            RAISE_APPLICATION_ERROR(-20011,
                'Salário incompatível com o cargo ' || p_cargo);
        END IF;

        -- Insere o funcionário
        INSERT INTO funcionarios (id, nome, salario, cargo, depto_id)
        VALUES (seq_funcionarios.NEXTVAL, p_nome, p_salario, p_cargo, p_depto_id)
        RETURNING id INTO v_id;

        -- Atualiza variável do pacote (persiste na sessão!)
        v_ultima_contratacao := p_nome;

        COMMIT;

        DBMS_OUTPUT.PUT_LINE('✓ Funcionário contratado: ' || p_nome || ' (ID=' || v_id || ')');
    END contratar;

    PROCEDURE demitir(
        p_id     NUMBER,
        p_motivo VARCHAR2 DEFAULT 'Demissão sem justa causa'
    )
    IS
        v_nome VARCHAR2(100);
    BEGIN
        SELECT nome INTO v_nome FROM funcionarios WHERE id = p_id;

        -- Move para histórico antes de excluir
        INSERT INTO historico_funcionarios (func_id, nome, data_saida, motivo)
        VALUES (p_id, v_nome, SYSDATE, p_motivo);

        DELETE FROM funcionarios WHERE id = p_id;

        COMMIT;

        DBMS_OUTPUT.PUT_LINE('✓ Funcionário ' || v_nome || ' desligado. Motivo: ' || p_motivo);
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            DBMS_OUTPUT.PUT_LINE('✗ Funcionário com ID ' || p_id || ' não encontrado.');
    END demitir;

    PROCEDURE promover(
        p_id           NUMBER,
        p_novo_cargo   VARCHAR2,
        p_aumento_pct  NUMBER
    )
    IS
        v_nome    VARCHAR2(100);
        v_cargo_antigo VARCHAR2(50);
    BEGIN
        SELECT nome, cargo INTO v_nome, v_cargo_antigo
        FROM funcionarios WHERE id = p_id;

        UPDATE funcionarios
        SET cargo = p_novo_cargo,
            salario = salario * (1 + p_aumento_pct / 100)
        WHERE id = p_id;

        COMMIT;

        DBMS_OUTPUT.PUT_LINE('✓ ' || v_nome || ' promovido de ' ||
            v_cargo_antigo || ' para ' || p_novo_cargo || ' com ' ||
            p_aumento_pct || '% de aumento.');
    END promover;

    -- ============ IMPLEMENTAÇÃO DAS FUNCTIONS ============

    FUNCTION calcular_ferias(
        p_salario        NUMBER,
        p_meses_trabalho NUMBER
    ) RETURN NUMBER
    IS
        v_valor_ferias NUMBER;
        v_meses        NUMBER;
    BEGIN
        v_meses := LEAST(p_meses_trabalho, 12);
        v_valor_ferias := (p_salario / 12) * v_meses;

        -- Adiciona 1/3 constitucional
        v_valor_ferias := v_valor_ferias + (v_valor_ferias / 3);

        RETURN ROUND(v_valor_ferias, 2);
    END calcular_ferias;

    FUNCTION validar_salario_cargo(
        p_cargo   VARCHAR2,
        p_salario NUMBER
    ) RETURN BOOLEAN
    IS
    BEGIN
        CASE UPPER(p_cargo)
            WHEN 'ESTAGIÁRIO'  THEN RETURN p_salario BETWEEN 800  AND 2000;
            WHEN 'ASSISTENTE'  THEN RETURN p_salario BETWEEN 1518 AND 3500;
            WHEN 'ANALISTA'    THEN RETURN p_salario BETWEEN 3000 AND 8000;
            WHEN 'GERENTE'     THEN RETURN p_salario BETWEEN 7000 AND 20000;
            WHEN 'DIRETOR'     THEN RETURN p_salario >= 15000;
            ELSE RETURN TRUE; -- cargo desconhecido, aceita qualquer valor
        END CASE;
    END validar_salario_cargo;

    FUNCTION total_por_depto(
        p_depto_id NUMBER
    ) RETURN NUMBER
    IS
        v_total NUMBER := 0;
    BEGIN
        SELECT COUNT(*) INTO v_total
        FROM funcionarios
        WHERE depto_id = p_depto_id;

        RETURN v_total;
    END total_por_depto;

END pkg_rh;
/

Chamando Procedures e Functions do Pacote


-- Chamando procedures
EXEC pkg_rh.contratar('Carlos Oliveira', 5200.00, 'Analista', 10);
EXEC pkg_rh.contratar('Marina Souza', 9000.00, 'Gerente', 10);
EXEC pkg_rh.promover(1, 'Coordenador', 15);
EXEC pkg_rh.demitir(5, 'Pedido de demissão');

-- Chamando functions
SELECT nome, salario, pkg_rh.calcular_ferias(salario, 12) AS ferias
FROM funcionarios;

-- Functions em condições
SELECT * FROM funcionarios
WHERE pkg_rh.total_por_depto(depto_id) > 3;

-- Acessando constante do pacote
SELECT pkg_rh.c_salario_minimo FROM DUAL;

Sobrecarga de Subprogramas (Overloading)

O Oracle permite que um pacote tenha múltiplas procedures ou functions com o mesmo nome, desde que tenham assinaturas diferentes (número ou tipo de parâmetros): 


-- Especificação com sobrecarga
CREATE OR REPLACE PACKAGE pkg_util AS

    -- Versão 1: aceita NUMBER

A procedure deve receber o p_depto_id (IN) e retornar p_media (OUT NUMBER) e p_contagem (OUT NUMBER) com a média salarial e quantidade de funcionários do departamento.

Gabarito: 

CREATE OR REPLACE PROCEDURE media_salarial_depto(
    p_depto_id  IN  NUMBER,
    p_media     OUT NUMBER,
    p_contagem  OUT NUMBER
)
IS
BEGIN
    SELECT AVG(salario), COUNT(*)
    INTO p_media, p_contagem
    FROM funcionarios
    WHERE depto_id = p_depto_id;

    IF p_contagem = 0 THEN
        p_media := 0;
        DBMS_OUTPUT.PUT_LINE('Departamento vazio ou inexistente.');
    END IF;
END media_salarial_depto;

CREATE OR REPLACE FUNCTION tempo_empresa_func(

3. Crie uma procedure com SAVEPOINT que faz transferência entre contas com validação

Considere uma tabela contas(id, titular, saldo). A procedure deve debitar de uma conta, creditar em outra, usar SAVEPOINT para garantir que nenhuma conta fique com saldo negativo, e validar que o valor da transferência é maior que zero.

Gabarito: 

-- Tabela de contas
CREATE TABLE contas (
    id      NUMBER PRIMARY KEY,
    titular VARCHAR2(100) NOT NULL,
    saldo   NUMBER(12,2) DEFAULT 0 CHECK (saldo >= 0)
);

INSERT INTO contas VALUES (1, 'João Silva', 10000);
INSERT INTO contas VALUES (2, 'Maria Santos', 5000);
INSERT INTO contas VALUES (3, 'Pedro Costa', 2000);

CREATE OR REPLACE PROCEDURE transferencia_bancaria(
    p_conta_origem  NUMBER,
    p_conta_destino NUMBER,
    p_valor         NUMBER
)
IS
    v_saldo_origem  NUMBER;
    v_saldo_destino NUMBER;
    v_titular_origem VARCHAR2(100);
    v_titular_destino VARCHAR2(100);
BEGIN
    -- Validação: valor deve ser positivo
    IF p_valor <= 0 THEN
        RAISE_APPLICATION_ERROR(-20101, 'Valor da transferência deve ser maior que zero.');
    END IF;

    -- Validação: contas diferentes
    IF p_conta_origem = p_conta_destino THEN
        RAISE_APPLICATION_ERROR(-20102, 'Origem e destino não podem ser a mesma conta.');
    END IF;

    -- Obtém dados da conta origem
    SELECT titular, saldo INTO v_titular_origem, v_saldo_origem
    FROM contas WHERE id = p_conta_origem FOR UPDATE;

    -- Verifica saldo
    IF v_saldo_origem < p_valor THEN
        RAISE_APPLICATION_ERROR(-20103,
            'Saldo insuficiente. Saldo atual: R$ ' ||
            TO_CHAR(v_saldo_origem, 'FM999G990D00'));
    END IF;

    -- Obtém dados da conta destino
    SELECT titular, saldo INTO v_titular_destino, v_saldo_destino
    FROM contas WHERE id = p_conta_destino FOR UPDATE;

    -- Salva ponto de controle
    SAVEPOINT antes_operacao;

    -- Debita origem
    UPDATE contas SET saldo = saldo - p_valor WHERE id = p_conta_origem;

    -- Verifica se origem ficou negativa
    SELECT saldo INTO v_saldo_origem FROM contas WHERE id = p_conta_origem;

    IF v_saldo_origem < 0 THEN
        ROLLBACK TO antes_operacao;
        RAISE_APPLICATION_ERROR(-20104, 'Erro: conta origem ficaria negativa.');
    END IF;

    -- Credita destino
    UPDATE contas SET saldo = saldo + p_valor WHERE id = p_conta_destino;

    -- Verifica se destino está ok (limite de depósito, por exemplo)
    SELECT saldo INTO v_saldo_destino FROM contas WHERE id = p_conta_destino;

    IF v_saldo_destino > 1000000 THEN
        ROLLBACK TO antes_operacao;
        RAISE_APPLICATION_ERROR(-20105,
            'Erro: conta destino excederia o limite de R$ 1.000.000,00.');
    END IF;

    COMMIT;

    DBMS_OUTPUT.PUT_LINE('✓ Transferência de R$ ' ||
        TO_CHAR(p_valor, 'FM999G990D00') || ' realizada:');
    DBMS_OUTPUT.PUT_LINE('  De: ' || v_titular_origem || ' (saldo: R$ ' ||
        TO_CHAR(v_saldo_origem, 'FM999G990D00') || ')');
    DBMS_OUTPUT.PUT_LINE('  Para: ' || v_titular_destino || ' (saldo: R$ ' ||
        TO_CHAR(v_saldo_destino, 'FM999G990D00') || ')');

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('✗ Conta de origem ou destino não encontrada.');
    WHEN OTHERS THEN
        ROLLBACK;
        DBMS_OUTPUT.PUT_LINE('✗ Erro na transferência: ' || SQLERRM);
        RAISE;
END transferencia_bancaria;
/

-- Testando
EXEC transferencia_bancaria(1, 2, 1500.00);

4. Crie um package com 2 procedures e 1 function relacionadas à folha de pagamento

Crie o package pkg_folha contendo:

  • calcular_salario_liquido(p_id NUMBER) RETURN NUMBER — function que desconta INSS e IR do salário bruto
  • gerar_holerite(p_id NUMBER) — procedure que exibe (DBMS_OUTPUT) o holerite completo do funcionário
  • reajustar_por_cargo(p_cargo VARCHAR2, p_percentual NUMBER) — procedure que reajusta todos os funcionários de um cargo
Gabarito: 

CREATE OR REPLACE PACKAGE pkg_folha AS

    FUNCTION calcular_salario_liquido(p_id NUMBER) RETURN NUMBER;

    PROCEDURE gerar_holerite(p_id NUMBER);

    PROCEDURE reajustar_por_cargo(p_cargo VARCHAR2, p_percentual NUMBER);

END pkg_folha;
/

CREATE OR REPLACE PACKAGE BODY pkg_folha AS

    -- Função privada para calcular INSS
    FUNCTION calc_inss(p_salario NUMBER) RETURN NUMBER
    IS
    BEGIN
        IF p_salario <= 1518.00 THEN
            RETURN p_salario * 0.075;
        ELSIF p_salario <= 2793.88 THEN
            RETURN p_salario * 0.09;
        ELSIF p_salario <= 4190.83 THEN
            RETURN p_salario * 0.12;
        ELSE
            RETURN p_salario * 0.14;
        END IF;
    END calc_inss;

    -- Função privada para calcular IR
    FUNCTION calc_ir(p_salario_base NUMBER) RETURN NUMBER
    IS
        v_base NUMBER := p_salario_base;
    BEGIN
        IF v_base <= 2259.20 THEN
            RETURN 0;
        ELSIF v_base <= 2826.65 THEN
            RETURN v_base * 0.075;
        ELSIF v_base <= 3751.05 THEN
            RETURN v_base * 0.15;
        ELSIF v_base <= 4664.68 THEN
            RETURN v_base * 0.225;
        ELSE
            RETURN v_base * 0.275;
        END IF;
    END calc_ir;

    FUNCTION calcular_salario_liquido(p_id NUMBER) RETURN NUMBER
    IS
        v_salario_bruto NUMBER;
        v_inss          NUMBER;
        v_ir            NUMBER;
        v_liquido       NUMBER;
    BEGIN
        SELECT salario INTO v_salario_bruto
        FROM funcionarios WHERE id = p_id;

        v_inss    := calc_inss(v_salario_bruto);
        v_ir      := calc_ir(v_salario_bruto - v_inss);
        v_liquido := v_salario_bruto - v_inss - v_ir;

        RETURN ROUND(v_liquido, 2);
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RETURN NULL;
    END calcular_salario_liquido;

    PROCEDURE gerar_holerite(p_id NUMBER)
    IS
        v_nome          VARCHAR2(100);
        v_cargo         VARCHAR2(50);
        v_salario_bruto NUMBER;
        v_inss          NUMBER;
        v_ir            NUMBER;
        v_liquido       NUMBER;
    BEGIN
        SELECT nome, cargo, salario INTO v_nome, v_cargo, v_salario_bruto
        FROM funcionarios WHERE id = p_id;

        v_inss    := ROUND(calc_inss(v_salario_bruto), 2);
        v_ir      := ROUND(calc_ir(v_salario_bruto - v_inss), 2);
        v_liquido := v_salario_bruto - v_inss - v_ir;

        DBMS_OUTPUT.PUT_LINE('============================================');
        DBMS_OUTPUT.PUT_LINE('          HOLERITE DE PAGAMENTO');
        DBMS_OUTPUT.PUT_LINE('============================================');
        DBMS_OUTPUT.PUT_LINE('Funcionário : ' || v_nome);
        DBMS_OUTPUT.PUT_LINE('Cargo       : ' || v_cargo);
        DBMS_OUTPUT.PUT_LINE('--------------------------------------------');
        DBMS_OUTPUT.PUT_LINE('Salário Bruto : R$ ' || TO_CHAR(v_salario_bruto, 'FM999G990D00'));
        DBMS_OUTPUT.PUT_LINE('INSS (-)      : R$ ' || TO_CHAR(v_inss, 'FM999G990D00'));
        DBMS_OUTPUT.PUT_LINE('IR (-)        : R$ ' || TO_CHAR(v_ir, 'FM999G990D00'));
        DBMS_OUTPUT.PUT_LINE('--------------------------------------------');
        DBMS_OUTPUT.PUT_LINE('Salário Líquido: R$ ' || TO_CHAR(v_liquido, 'FM999G990D00'));
        DBMS_OUTPUT.PUT_LINE('============================================');
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            DBMS_OUTPUT.PUT_LINE('Funcionário não encontrado.');
    END gerar_holerite;

    PROCEDURE reajustar_por_cargo(p_cargo VARCHAR2, p_percentual NUMBER)
    IS
        v_contagem NUMBER;
    BEGIN
        UPDATE funcionarios
        SET salario = salario * (1 + p_percentual / 100)
        WHERE UPPER(cargo) = UPPER(p_cargo);

        v_contagem := SQL%ROWCOUNT;

        COMMIT;

        DBMS_OUTPUT.PUT_LINE('✓ ' || v_contagem || ' funcionário(s) do cargo ' ||
            p_cargo || ' reajustado(s) em ' || p_percentual || '%.');
    END reajustar_por_cargo;

END pkg_folha;
/

-- Testando
EXEC pkg_folha.gerar_holerite(1);
SELECT nome, salario, pkg_folha.calcular_salario_liquido(id) AS liquido FROM funcionarios;
EXEC pkg_folha.reajustar_por_cargo('Analista', 5);
