PL/pgSQL — Conceito de Bloco

PL/pgSQL (Procedural Language/PostgreSQL) é a linguagem procedural nativa do PostgreSQL. Ela estende o SQL puro com construções de programação como variáveis, condicionais, loops, cursores e tratamento de exceções. Todo código PL/pgSQL é estruturado em blocos.

Estrutura de um Bloco PL/pgSQL


-- Estrutura básica de um bloco PL/pgSQL
[ <> ]
[ DECLARE
    -- declaração de variáveis (opcional)
]
BEGIN
    -- código SQL e PL/pgSQL (obrigatório)
[ EXCEPTION
    -- tratamento de exceções (opcional)
]
END [ rótulo ];

Blocos podem ser aninhados: é possível declarar sub-blocos dentro de um bloco principal. Cada sub-bloco tem seu próprio escopo de variáveis e seu próprio EXCEPTION.

Primeiro Exemplo: Bloco Anônimo (DO)


-- Bloco anônimo PL/pgSQL (executado uma única vez, sem ser armazenado)
DO $$
DECLARE
    v_nome VARCHAR(100);
    v_idade INTEGER;
BEGIN
    v_nome := 'João Silva';
    v_idade := 30;

    RAISE NOTICE 'Cliente: %, Idade: % anos', v_nome, v_idade;
END;
$$;

Função Armazenada: Estrutura Completa


-- Função armazenada com parâmetros e retorno
CREATE OR REPLACE FUNCTION calcular_desconto(
    p_preco NUMERIC,
    p_percentual NUMERIC DEFAULT 10
)
RETURNS NUMERIC
LANGUAGE plpgsql
AS $$
DECLARE
    v_valor_desconto NUMERIC;
    v_preco_final NUMERIC;
BEGIN
    v_valor_desconto := p_preco * (p_percentual / 100.0);
    v_preco_final := p_preco - v_valor_desconto;

    RAISE NOTICE 'Preço original: %, Desconto: %%, Valor final: %',
        p_preco, p_percentual, v_preco_final;

    RETURN v_preco_final;
END;
$$;

-- Chamar a função
SELECT calcular_desconto(1000, 15);
SELECT calcular_desconto(500);  -- usa DEFAULT 10%

Blocos Aninhados com Escopo de Variáveis


DO $$
DECLARE
    v_externa VARCHAR := 'Variável do bloco externo';

%TYPE — Tipo baseado em coluna de tabela

O atributo %TYPE declara uma variável com o mesmo tipo de uma coluna de tabela. Isso torna o código mais robusto: se o tipo da coluna mudar, a variável acompanha automaticamente.


-- Criar tabela de exemplo
CREATE TABLE clientes (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100),
    saldo NUMERIC(12,2) DEFAULT 0,
    data_nascimento DATE
);

DO $$
DECLARE
    -- %TYPE: tipo igual à coluna da tabela
    v_nome clientes.nome%TYPE;
    v_email clientes.email%TYPE;
    v_saldo clientes.saldo%TYPE;
BEGIN
    -- SELECT INTO: atribui resultado da query à variável
    SELECT nome, email, saldo
    INTO v_nome, v_email, v_saldo
    FROM clientes
    WHERE id = 1;

    IF FOUND THEN
        RAISE NOTICE 'Cliente: %, Email: %, Saldo: R$ %', v_nome, v_email, v_saldo;
    ELSE
        RAISE NOTICE 'Nenhum cliente encontrado com id = 1';
    END IF;
END;
$$;

%ROWTYPE — Linha inteira da tabela

O atributo %ROWTYPE declara uma variável do tipo record que corresponde a uma linha completa da tabela. Cada coluna pode ser acessada como variavel.nome_coluna.


DO $$
DECLARE
    -- %ROWTYPE: variável tipo "linha completa"
    v_cliente clientes%ROWTYPE;
BEGIN
    -- SELECT INTO com %ROWTYPE
    SELECT * INTO v_cliente FROM clientes WHERE id = 1;

    RAISE NOTICE 'ID: %, Nome: %, Email: %, Saldo: R$ %',
        v_cliente.id, v_cliente.nome, v_cliente.email, v_cliente.saldo;
END;
$$;

SELECT INTO — Atribuindo resultados a variáveis


DO $$
DECLARE
    v_count INTEGER;
    v_total_saldo NUMERIC(12,2);
    v_media NUMERIC(12,2);
BEGIN
    -- Atribuir resultado de agregação
    SELECT COUNT(*), COALESCE(SUM(saldo), 0), COALESCE(AVG(saldo), 0)
    INTO v_count, v_total_saldo, v_media
    FROM clientes;

    RAISE NOTICE 'Total de clientes: %, Saldo total: R$ %, Média: R$ %',
        v_count, v_total_saldo, v_media;

    -- STRICT: gera erro se zero ou mais de uma linha retornar
    -- SELECT nome INTO STRICT v_nome FROM clientes WHERE id = 999;
    -- Isso geraria NO_DATA_FOUND
END;
$$;

FOUND — Variável booleana especial

Após um SELECT INTO, INSERT, UPDATE ou DELETE, a variável implícita FOUND indica se a última operação afetou alguma linha.


DO $$
DECLARE
    v_nome clientes.nome%TYPE;
BEGIN
    SELECT nome INTO v_nome FROM clientes WHERE id = 999;

    IF NOT FOUND THEN
        RAISE NOTICE 'Cliente ID 999 não encontrado!';
    END IF;

    UPDATE clientes SET saldo = 0 WHERE id = 999;

    IF NOT FOUND THEN
        RAISE NOTICE 'Nenhuma linha foi atualizada!';
    END IF;
END;
$$;

Estruturas Condicionais

IF / THEN / ELSIF / ELSE


DO $$
DECLARE
    v_nota NUMERIC := 7.5;
    v_conceito VARCHAR(20);
BEGIN
    IF v_nota >= 9.0 THEN
        v_conceito := 'Excelente';
    ELSIF v_nota >= 8.0 THEN
        v_conceito := 'Ótimo';
    ELSIF v_nota >= 7.0 THEN
        v_conceito := 'Bom';
    ELSIF v_nota >= 6.0 THEN
        v_conceito := 'Regular';
    ELSIF v_nota >= 5.0 THEN
        v_conceito := 'Suficiente';
    ELSE
        v_conceito := 'Insuficiente';
    END IF;

    RAISE NOTICE 'Nota: % → Conceito: %', v_nota, v_conceito;
END;
$$;

Exemplo: IF com múltiplas condições e operadores lógicos


CREATE OR REPLACE FUNCTION classificar_cliente(p_id INTEGER)
RETURNS VARCHAR
LANGUAGE plpgsql
AS $$
DECLARE
    v_saldo clientes.saldo%TYPE;
    v_status clientes.status%TYPE;
    v_classificacao VARCHAR(50);
BEGIN
    SELECT saldo, status INTO v_saldo, v_status
    FROM clientes WHERE id = p_id;

    IF NOT FOUND THEN
        RETURN 'Cliente não encontrado';
    END IF;

    IF v_status = 'inativo' THEN
        v_classificacao := 'Inativo';
    ELSIF v_saldo >= 5000 AND v_status = 'ativo' THEN
        v_classificacao := 'VIP';
    ELSIF v_saldo >= 2000 AND v_status = 'ativo' THEN
        v_classificacao := 'Premium';
    ELSIF v_saldo >= 0 AND v_status = 'ativo' THEN
        v_classificacao := 'Regular';
    ELSE
        v_classificacao := 'Não classificado';
    END IF;

    RETURN v_classificacao;
END;
$$;

-- Testar
SELECT id, nome, classificar_cliente(id) AS classificacao FROM clientes;

CASE — Duas formas

Forma 1: CASE simples (compara um valor)


DO $$
DECLARE
    v_dia_semana INTEGER := EXTRACT(DOW FROM CURRENT_DATE);
    v_nome_dia VARCHAR(20);
BEGIN
    CASE v_dia_semana
        WHEN 0 THEN v_nome_dia := 'Domingo';
        WHEN 1 THEN v_nome_dia := 'Segunda-feira';
        WHEN 2 THEN v_nome_dia := 'Terça-feira';
        WHEN 3 THEN v_nome_dia := 'Quarta-feira';
        WHEN 4 THEN v_nome_dia := 'Quinta-feira';
        WHEN 5 THEN v_nome_dia := 'Sexta-feira';
        WHEN 6 THEN v_nome_dia := 'Sábado';
    END CASE;

    RAISE NOTICE 'Hoje é %', v_nome_dia;
END;
$$;

Forma 2: CASE pesquisado (avalia condições)


DO $$
DECLARE
    v_hora INTEGER := EXTRACT(HOUR FROM now());
    v_periodo VARCHAR(20);
BEGIN
    CASE
        WHEN v_hora >= 5 AND v_hora < 12 THEN
            v_periodo := 'Manhã';
        WHEN v_hora >= 12 AND v_hora < 18 THEN
            v_periodo := 'Tarde';
        WHEN v_hora >= 18 AND v_hora < 23 THEN
            v_periodo := 'Noite';
        ELSE
            v_periodo := 'Madrugada';
    END CASE;

    RAISE NOTICE 'Hora: % → Período: %', v_hora, v_periodo;
END;
$$;

CASE como expressão (dentro de SQL e atribuições)


DO $$
DECLARE
    v_saldo NUMERIC := 3500;

$$;

Estruturas de Repetição

LOOP com EXIT

O LOOP básico executa indefinidamente até encontrar um EXIT.


DO $$
DECLARE
    v_contador INTEGER := 1;
    v_soma INTEGER := 0;
BEGIN
    LOOP
        v_soma := v_soma + v_contador;
        v_contador := v_contador + 1;

        EXIT WHEN v_contador > 10;
    END LOOP;

    RAISE NOTICE 'Soma de 1 a 10: %', v_soma;
END;
$$;

LOOP com EXIT aninhado (rótulos)


DO $$
DECLARE
    v_i INTEGER;
    v_j INTEGER;
BEGIN
    <>
    FOR v_i IN 1..5 LOOP
        <>
        FOR v_j IN 1..5 LOOP
            IF v_i * v_j > 10 THEN
                RAISE NOTICE 'Saindo do loop externo: i=%, j=%, produto=%', v_i, v_j, v_i*v_j;
                EXIT loop_externo;  -- sai de ambos os loops
            END IF;
            RAISE NOTICE 'i=%, j=%, produto=%', v_i, v_j, v_i*v_j;
        END LOOP;
    END LOOP;
END;
$$;

WHILE


DO $$
DECLARE
    v_numero INTEGER := 100;
    v_passos INTEGER := 0;
BEGIN
    WHILE v_numero > 1 LOOP
        IF v_numero % 2 = 0 THEN
            v_numero := v_numero / 2;
        ELSE
            v_numero := v_numero * 3 + 1;
        END IF;

        v_passos := v_passos + 1;
        RAISE NOTICE 'Passo %: número = %', v_passos, v_numero;
    END LOOP;

    RAISE NOTICE 'Conjectura de Collatz: % passos para chegar a 1', v_passos;
END;
$$;

FOR Numérico (com range)


DO $$
DECLARE
    v_fatorial BIGINT := 1;
BEGIN
    FOR i IN 1..10 LOOP
        v_fatorial := v_fatorial * i;
        RAISE NOTICE '%! = %', i, v_fatorial;
    END LOOP;
END;
$$;

-- FOR reverso (REVERSE)
DO $$
BEGIN
    RAISE NOTICE 'Contagem regressiva:';
    FOR i IN REVERSE 10..1 LOOP
        RAISE NOTICE '%...', i;
    END LOOP;
    RAISE NOTICE 'Feliz Ano Novo!';
END;
$$;

-- FOR com step (BY)
DO $$
BEGIN
    RAISE NOTICE 'Números pares de 2 a 20:';
    FOR i IN 2..20 BY 2 LOOP
        RAISE NOTICE '%', i;
    END LOOP;
END;
$$;

FOR com Query (Cursor Loop implícito)


DO $$
DECLARE
    v_registro RECORD;
BEGIN
    -- Itera sobre cada linha retornada pela query
    FOR v_registro IN
        SELECT nome, email, saldo
        FROM clientes
        WHERE status = 'ativo'
        ORDER BY saldo DESC
    LOOP
        RAISE NOTICE 'Cliente: %, Email: %, Saldo: R$ %',
            v_registro.nome, v_registro.email, v_registro.saldo;
    END LOOP;

    -- Verifica se o loop iterou sobre pelo menos uma linha
    IF NOT FOUND THEN
        RAISE NOTICE 'Nenhum cliente ativo encontrado.';
    END IF;
END;
$$;

CONTINUE — Pular iteração


DO $$
BEGIN
    RAISE NOTICE 'Números ímpares de 1 a 20:';
    FOR i IN 1..20 LOOP
        -- Pula números pares
        CONTINUE WHEN i % 2 = 0;

        RAISE NOTICE '%', i;
    END LOOP;
END;
$$;

Todos os tipos de loop — Resumo prático


CREATE OR REPLACE FUNCTION demonstrar_loops()
RETURNS TEXT
LANGUAGE plpgsql
AS $$
DECLARE
    v_resultado TEXT := '';
BEGIN
    -- LOOP com EXIT
    v_resultado := v_resultado || 'LOOP: ';
    DECLARE
        c INTEGER := 0;
    BEGIN
        LOOP
            c := c + 1;
            EXIT WHEN c > 3;
            v_resultado := v_resultado || c || ' ';
        END LOOP;
    END;

    -- WHILE
    v_resultado := v_resultado || E'\nWHILE: ';
    DECLARE
        c INTEGER := 1;
    BEGIN
        WHILE c <= 3 LOOP
            v_resultado := v_resultado || c || ' ';
            c := c + 1;
        END LOOP;
    END;

    -- FOR numérico
    v_resultado := v_resultado || E'\nFOR: ';
    FOR i IN 1..3 LOOP
        v_resultado := v_resultado || i || ' ';
    END LOOP;

    -- FOR reverso
    v_resultado := v_resultado || E'\nFOR REVERSE: ';
    FOR i IN REVERSE 3..1 LOOP
        v_resultado := v_resultado || i || ' ';
    END LOOP;

    RETURN v_resultado;
END;
$$;

Cursores em PL/pgSQL

Um cursor permite percorrer linha a linha o resultado de uma query, mantendo controle sobre a posição atual. Cursores são essenciais quando você precisa processar grandes volumes de dados de forma controlada, sem carregar tudo em memória de uma vez.

Declarar, Abrir, Fetch e Fechar — Passo a passo


DO $$
DECLARE
    -- 1. Declarar o cursor
    cur_clientes CURSOR FOR
        SELECT id, nome, email, saldo
        FROM clientes
        WHERE status = 'ativo'
        ORDER BY saldo DESC;

    -- Variáveis para receber os dados
    v_id clientes.id%TYPE;
    v_nome clientes.nome%TYPE;
    v_email clientes.email%TYPE;
    v_saldo clientes.saldo%TYPE;
BEGIN
    -- 2. Abrir o cursor
    OPEN cur_clientes;

    LOOP
        -- 3. Buscar próxima linha (FETCH)
        FETCH cur_clientes INTO v_id, v_nome, v_email, v_saldo;

        -- Sair quando não houver mais linhas
        EXIT WHEN NOT FOUND;

        RAISE NOTICE 'ID: %, Nome: %, Email: %, Saldo: R$ %',
            v_id, v_nome, v_email, v_saldo;
    END LOOP;

    -- 4. Fechar o cursor
    CLOSE cur_clientes;
END;
$$;

Cursor com parâmetros


DO $$
DECLARE
    -- Cursor que aceita parâmetros
    cur_por_saldo CURSOR (p_saldo_min NUMERIC, p_status VARCHAR) FOR
        SELECT id, nome, saldo
        FROM clientes
        WHERE saldo >= p_saldo_min
          AND status = p_status
        ORDER BY saldo DESC;

    v_registro RECORD;
BEGIN
    -- Abrir com parâmetros específicos
    OPEN cur_por_saldo(2000, 'ativo');

    LOOP
        FETCH cur_por_saldo INTO v_registro;
        EXIT WHEN NOT FOUND;

        RAISE NOTICE '% — Saldo: R$ %', v_registro.nome, v_registro.saldo;
    END LOOP;

    CLOSE cur_por_saldo;
END;
$$;

Cursor FOR Loop (forma simplificada)

O Cursor FOR Loop é a maneira mais simples e segura de iterar sobre um cursor — ele automaticamente abre, faz fetch e fecha o cursor.


DO $$
DECLARE
    v_registro RECORD;
    v_contador INTEGER := 0;
    v_soma_saldo NUMERIC := 0;
BEGIN
    -- FOR loop com cursor implícito (abre e fecha automaticamente)
    FOR v_registro IN
        SELECT nome, saldo
        FROM clientes
        WHERE status = 'ativo'
        ORDER BY saldo DESC
    LOOP
        v_contador := v_contador + 1;
        v_soma_saldo := v_soma_saldo + v_registro.saldo;

        RAISE NOTICE '%° — %: R$ %',
            v_contador, v_registro.nome, v_registro.saldo;
    END LOOP;

    RAISE NOTICE 'Total: % clientes, Saldo total: R$ %', v_contador, v_soma_saldo;
END;
$$;

Exemplo completo: Processamento com cursor e atualização condicional


CREATE OR REPLACE FUNCTION atualizar_classificacao_clientes()
RETURNS TABLE(nome VARCHAR, saldo NUMERIC, classificacao VARCHAR)
LANGUAGE plpgsql
AS $$
DECLARE
    cur CURSOR FOR
        SELECT id, nome, saldo, status
        FROM clientes
        WHERE status = 'ativo'
        FOR UPDATE;  -- bloqueia linhas para atualização

    v_registro RECORD;
    v_nova_classificacao VARCHAR(20);
BEGIN
    OPEN cur;

    LOOP
        FETCH cur INTO v_registro;
        EXIT WHEN NOT FOUND;

        -- Classificar com base no saldo
        v_nova_classificacao := CASE
            WHEN v_registro.saldo >= 5000 THEN 'VIP'
            WHEN v_registro.saldo >= 2000 THEN 'Premium'
            ELSE 'Regular'
        END;

        -- Atualizar a linha atual do cursor
        UPDATE clientes
        SET status = v_nova_classificacao
        WHERE CURRENT OF cur;

        -- Retornar a linha processada
        nome := v_registro.nome;
        saldo := v_registro.saldo;
        classificacao := v_nova_classificacao;
        RETURN NEXT;
    END LOOP;

    CLOSE cur;
END;
$$;

-- Executar
SELECT * FROM atualizar_classificacao_clientes();

SCROLL CURSOR — Navegação bidimensional


DO $$
DECLARE
    -- SCROLL permite navegar para frente e para trás
    cur SCROLL CURSOR FOR
        SELECT nome, saldo FROM clientes ORDER BY id;

    v_registro RECORD;
BEGIN
    OPEN cur;

    -- Última linha
    FETCH LAST FROM cur INTO v_registro;
    RAISE NOTICE 'Último: % — R$ %', v_registro.nome, v_registro.saldo;

    -- Primeira linha
    FETCH FIRST FROM cur INTO v_registro;
    RAISE NOTICE 'Primeiro: % — R$ %', v_registro.nome, v_registro.saldo;

    -- Linha anterior (ABSOLUTE -1 relativo à posição atual)
    FETCH ABSOLUTE 2 FROM cur INTO v_registro;
    RAISE NOTICE 'Posição 2: % — R$ %', v_registro.nome, v_registro.saldo;

    -- Voltar uma posição
    FETCH PRIOR FROM cur INTO v_registro;
    RAISE NOTICE 'Anterior: % — R$ %', v_registro.nome, v_registro.saldo;

    CLOSE cur;
END;
$$;

Triggers no PostgreSQL

Triggers (gatilhos) são funções especiais que o PostgreSQL executa automaticamente em resposta a eventos DML (INSERT, UPDATE, DELETE, TRUNCATE) em uma tabela. Diferente das Regras (que reescrevem a query), triggers são disparados durante a execução.

Anatomia de um Trigger: Função + Definição

Um trigger no PostgreSQL é sempre composto por duas partes:

  1. Trigger Function: uma função que retorna TRIGGER (escrita em PL/pgSQL ou outra linguagem procedural)
  2. CREATE TRIGGER: comando que associa a função a uma tabela e define quando/quem dispara

Estrutura básica de uma Trigger Function


CREATE OR REPLACE FUNCTION nome_funcao_trigger()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    -- variáveis (opcional)
BEGIN
    -- Lógica do trigger

    RETURN NEW;  -- para BEFORE INSERT/UPDATE (ou OLD para DELETE)
    -- ou RETURN NULL para cancelar a operação (BEFORE trigger apenas)
END;
$$;

CREATE TRIGGER — Sintaxe


CREATE [ OR REPLACE ] [ CONSTRAINT ] TRIGGER nome_trigger
    { BEFORE | AFTER | INSTEAD OF }
    { INSERT | UPDATE | DELETE | TRUNCATE }
    [ OR INSERT OR UPDATE OR DELETE OR TRUNCATE ]
    ON tabela_ou_view
    [ FROM referenced_table_name ]
    [ NOT DEFERRABLE | [ DEFERRABLE ] [ INITIALLY IMMEDIATE | INITIALLY DEFERRED ] ]
    [ REFERENCING { { OLD | NEW } TABLE [ AS ] transition_table_name } ]
    [ FOR [ EACH ] { ROW | STATEMENT } ]
    [ WHEN ( condicao ) ]
    EXECUTE { FUNCTION | PROCEDURE } nome_funcao_trigger( argumentos )

Tipos de Trigger: BEFORE, AFTER, INSTEAD OF

Tipo Momento do disparo Usos comuns
BEFORE Antes da operação ser executada Validação de dados, modificação de NEW, prevenção de operação (RETURN NULL)
AFTER Depois da operação ser concluída Auditoria, logging, replicação, atualização de tabelas relacionadas
INSTEAD OF No lugar da operação (apenas em views) Tornar views complexas atualizáveis (alternativa às regras)

Níveis: FOR EACH ROW vs FOR EACH STATEMENT

Nível Quantas vezes dispara OLD/NEW disponíveis? Casos de uso
FOR EACH ROW Uma vez por linha afetada Sim Validação linha a linha, modificação de dados, cálculo por registro
FOR EACH STATEMENT Uma vez por comando SQL Não Logging de operação, notificações pós-operação, manutenção geral

Variáveis Especiais em Triggers (TG_*)


CREATE OR REPLACE FUNCTION debug_trigger_info()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    RAISE NOTICE '===== Informações do Trigger =====';
    RAISE NOTICE 'TG_NAME: %', TG_NAME;
    RAISE NOTICE 'TG_WHEN: %', TG_WHEN;
    RAISE NOTICE 'TG_OP: %', TG_OP;
    RAISE NOTICE 'TG_LEVEL: %', TG_LEVEL;
    RAISE NOTICE 'TG_RELID: %', TG_RELID;
    RAISE NOTICE 'TG_TABLE_NAME: %', TG_TABLE_NAME;
    RAISE NOTICE 'TG_TABLE_SCHEMA: %', TG_TABLE_SCHEMA;
    RAISE NOTICE 'TG_NARGS: %', TG_NARGS;
    RAISE NOTICE 'TG_ARGV: %', TG_ARGV;

    IF TG_OP = 'INSERT' THEN
        RAISE NOTICE 'NEW: id=%, nome=%', NEW.id, NEW.nome;
    ELSIF TG_OP = 'UPDATE' THEN
        RAISE NOTICE 'OLD: nome=%, NEW: nome=%', OLD.nome, NEW.nome;
    ELSIF TG_OP = 'DELETE' THEN
        RAISE NOTICE 'OLD: id=%, nome=%', OLD.id, OLD.nome;
    END IF;

    RETURN NEW;

Exemplo 1: BEFORE INSERT — Validação e modificação de dados


CREATE OR REPLACE FUNCTION trg_validar_cliente()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    -- Validação: nome não pode ser vazio
    IF NEW.nome IS NULL OR TRIM(NEW.nome) = '' THEN
        RAISE EXCEPTION 'O nome do cliente não pode ser vazio!';
    END IF;

    -- Validação: saldo não pode ser negativo
    IF NEW.saldo < 0 THEN
        RAISE EXCEPTION 'O saldo inicial não pode ser negativo!';
    END IF;

    -- Modificação automática: normalizar nome (primeira letra maiúscula)
    NEW.nome := INITCAP(TRIM(NEW.nome));

    -- Modificação automática: forçar email minúsculo
    IF NEW.email IS NOT NULL THEN
        NEW.email := LOWER(TRIM(NEW.email));
    END IF;

    -- Definir data de criação automaticamente
    NEW.criado_em := COALESCE(NEW.criado_em, now());
    NEW.atualizado_em := now();

    RETURN NEW;  -- permite a inserção com os dados modificados
END;
$$;

CREATE TRIGGER trigger_validar_cliente
    BEFORE INSERT ON clientes
    FOR EACH ROW
    EXECUTE FUNCTION trg_validar_cliente();

-- Tabela de log
CREATE TABLE log_alteracoes (
    id SERIAL PRIMARY KEY,
    tabela VARCHAR(50),
    operacao VARCHAR(10),
    registro_id INTEGER,
    dados_antigos JSONB,
    dados_novos JSONB,

    data_hora TIMESTAMP DEFAULT now()
);

-- Função trigger genérica de auditoria
CREATE OR REPLACE FUNCTION trg_auditar_clientes()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        INSERT INTO log_alteracoes (tabela, operacao, registro_id, dados_novos)
        VALUES (TG_TABLE_NAME, 'INSERT', NEW.id,
                to_jsonb(NEW));

    ELSIF TG_OP = 'UPDATE' THEN
        INSERT INTO log_alteracoes (tabela, operacao, registro_id, dados_antigos, dados_novos)
        VALUES (TG_TABLE_NAME, 'UPDATE', NEW.id,
                to_jsonb(OLD), to_jsonb(NEW));

    ELSIF TG_OP = 'DELETE' THEN
        INSERT INTO log_alteracoes (tabela, operacao, registro_id, dados_antigos)
        VALUES (TG_TABLE_NAME, 'DELETE', OLD.id,
                to_jsonb(OLD));

    END IF;

    RETURN NULL;  -- AFTER trigger: valor de retorno é ignorado
END;
$$;

    RETURN NEW;
END;
$$;

Passo a passo: Sistema de validação com trigger

Vamos construir um sistema completo de validação de pedidos usando triggers. O sistema impede pedidos com valor negativo, atualiza o estoque automaticamente e registra um log de todas as alterações.

Passo 1 — Criar tabelas


-- Tabela de produtos com controle de estoque
CREATE TABLE produtos (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    preco_unitario NUMERIC(10,2) NOT NULL CHECK (preco_unitario > 0),
    estoque INTEGER NOT NULL DEFAULT 0 CHECK (estoque >= 0),
    ativo BOOLEAN DEFAULT TRUE,
    criado_em TIMESTAMP DEFAULT now(),
    atualizado_em TIMESTAMP DEFAULT now()
);

-- Tabela de pedidos
CREATE TABLE pedidos (
    id SERIAL PRIMARY KEY,
    cliente_id INTEGER NOT NULL,
    data_pedido TIMESTAMP DEFAULT now(),
    valor_total NUMERIC(12,2) DEFAULT 0,
    status VARCHAR(20) DEFAULT 'pendente',
    atualizado_em TIMESTAMP DEFAULT now()
);

-- Tabela de itens do pedido
CREATE TABLE itens_pedido (
    id SERIAL PRIMARY KEY,
    pedido_id INTEGER REFERENCES pedidos(id) ON DELETE CASCADE,
    produto_id INTEGER REFERENCES produtos(id),
    quantidade INTEGER NOT NULL CHECK (quantidade > 0),
    preco_unitario NUMERIC(10,2) NOT NULL,
    subtotal NUMERIC(12,2) GENERATED ALWAYS AS (quantidade * preco_unitario) STORED
);

-- Tabela de log de operações
CREATE TABLE log_estoque (
    id SERIAL PRIMARY KEY,
    produto_id INTEGER,
    operacao VARCHAR(10),
    quantidade_anterior INTEGER,
    quantidade_nova INTEGER,
    pedido_id INTEGER,
    data_hora TIMESTAMP DEFAULT now()
);

Passo 2 — Trigger de validação de itens do pedido


CREATE OR REPLACE FUNCTION trg_validar_item_pedido()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    v_estoque_atual INTEGER;
    v_preco_produto NUMERIC(10,2);
    v_produto_ativo BOOLEAN;
    v_nome_produto VARCHAR(100);
BEGIN
    -- Busca informações do produto
    SELECT estoque, preco_unitario, ativo, nome
    INTO v_estoque_atual, v_preco_produto, v_produto_ativo, v_nome_produto
    FROM produtos
    WHERE id = NEW.produto_id;

    -- Verifica se o produto existe
    IF NOT FOUND THEN
        RAISE EXCEPTION 'Produto ID % não encontrado!', NEW.produto_id;
    END IF;

    -- Verifica se o produto está ativo
    IF NOT v_produto_ativo THEN
        RAISE EXCEPTION 'Produto "%" está desativado!', v_nome_produto;
    END IF;

    -- Verifica se há estoque suficiente
    IF TG_OP = 'INSERT' THEN
        IF v_estoque_atual < NEW.quantidade THEN
            RAISE EXCEPTION 'Estoque insuficiente para "%": solicitado %, disponível %',
                v_nome_produto, NEW.quantidade, v_estoque_atual;
        END IF;
    END IF;

    -- Preenche o preço unitário automaticamente
    NEW.preco_unitario := v_preco_produto;

    RETURN NEW;
END;
$$;

CREATE TRIGGER trigger_validar_item
    BEFORE INSERT ON itens_pedido
    FOR EACH ROW
    EXECUTE FUNCTION trg_validar_item_pedido();

Passo 3 — Trigger de atualização de estoque e log


CREATE OR REPLACE FUNCTION trg_baixar_estoque()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    v_estoque_anterior INTEGER;
BEGIN
    -- Registrar estoque antes
    SELECT estoque INTO v_estoque_anterior
    FROM produtos WHERE id = NEW.produto_id;

    -- Baixar estoque
    UPDATE produtos
    SET estoque = estoque - NEW.quantidade,
        atualizado_em = now()
    WHERE id = NEW.produto_id;

    -- Atualizar valor total do pedido
    UPDATE pedidos
    SET valor_total = valor_total + (NEW.quantidade * NEW.preco_unitario),
        atualizado_em = now()
    WHERE id = NEW.pedido_id;

    -- Registrar no log
    INSERT INTO log_estoque (produto_id, operacao, quantidade_anterior,
                              quantidade_nova, pedido_id)
    VALUES (NEW.produto_id, 'SAIDA', v_estoque_anterior,
            v_estoque_anterior - NEW.quantidade, NEW.pedido_id);

    RETURN NEW;
END;
$$;

CREATE TRIGGER trigger_baixar_estoque
    AFTER INSERT ON itens_pedido
    FOR EACH ROW
    EXECUTE FUNCTION trg_baixar_estoque();

Comparação PL/pgSQL (PostgreSQL) vs PL/SQL (Oracle)

Embora ambas sejam linguagens procedurais para bancos de dados, existem diferenças significativas na sintaxe e nos recursos disponíveis.

Característica PL/pgSQL (PostgreSQL) PL/SQL (Oracle)
Estrutura do bloco DECLARE ... BEGIN ... END; DECLARE ... BEGIN ... EXCEPTION ... END;
Packages Não tem packages nativos (usa schemas para agrupar) Packages (spec + body) — agrupam procedures, funções, variáveis e tipos
Assinatura de função RETURNS tipo AS $$ ... $$ LANGUAGE plpgsql; RETURN tipo IS ... BEGIN ... END;
Atribuição variavel := valor; variavel := valor;
SELECT INTO Atribui resultado a variáveis: SELECT col INTO var FROM... Atribui resultado a variáveis: SELECT col INTO var FROM...
%TYPE e %ROWTYPE Suportado Suportado
Cursores Declara, OPEN, FETCH, CLOSE; Cursor FOR loop Declara, OPEN, FETCH, CLOSE; Cursor FOR loop; cursores parametrizados
Exceções EXCEPTION WHEN condicao THEN ... (limitado) Sistema completo: PRAGMA EXCEPTION_INIT, RAISE_APPLICATION_ERROR, WHEN OTHERS, SQLERRM, SQLCODE
Triggers Função separada + CREATE TRIGGER; BEFORE/AFTER/INSTEAD OF Função inline ou separada; BEFORE/AFTER/INSTEAD OF; COMPOUND triggers (11g+)
Autonomous Transaction Não suporta nativamente (usa extensão dblink ou pg_background) PRAGMA AUTONOMOUS_TRANSACTION (transação independente dentro de outra)
Bulk Operations (FORALL, BULK COLLECT) Não tem equivalente direto (usa INSERT INTO ... SELECT para bulk) FORALL, BULK COLLECT INTO — operações em massa com performance otimizada
Collections (Arrays) ARRAY, array_agg(), unnest(). Arrays nativos simples VARRAY, NESTED TABLE, Associative Arrays (INDEX BY). Sistema rico de coleções
Overloading de funções Sim: mesmo nome, parâmetros diferentes Sim: mesmo nome, parâmetros diferentes (dentro de packages)
Execução dinâmica EXECUTE 'string_sql'; ou EXECUTE format(...) USING ... EXECUTE IMMEDIATE 'string_sql' USING ...;
Bloco anônimo DO $$ ... END; $$; BEGIN ... END;
Retorno de conjunto (SETOF) RETURNS SETOF tipo + RETURN NEXT ou RETURN QUERY RETURN SYS_REFCURSOR (aberto e retornado) ou PIPELINED

Exemplo lado a lado: Função que retorna total de pedidos por cliente

PL/pgSQL (PostgreSQL):


CREATE OR REPLACE FUNCTION total_pedidos_cliente(p_cliente_id INTEGER)
RETURNS TABLE(
    cliente_nome VARCHAR,
    total_pedidos BIGINT,
    valor_total NUMERIC
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
        SELECT c.nome, COUNT(p.id)::BIGINT, COALESCE(SUM(p.valor_total), 0)
        FROM clientes c
        LEFT JOIN pedidos p ON c.id = p.cliente_id
        WHERE c.id = p_cliente_id
        GROUP BY c.nome;
END;
$$;

PL/SQL (Oracle) — equivalente:


CREATE OR REPLACE FUNCTION total_pedidos_cliente(p_cliente_id NUMBER)
RETURN SYS_REFCURSOR
IS
    v_cursor SYS_REFCURSOR;
BEGIN
    OPEN v_cursor FOR
        SELECT c.nome, COUNT(p.id), NVL(SUM(p.valor_total), 0)
        FROM clientes c
        LEFT JOIN pedidos p ON c.id = p.cliente_id
        WHERE c.id = p_cliente_id
        GROUP BY c.nome;

    RETURN v_cursor;
END;

    departamento VARCHAR,
    total_funcionarios BIGINT,
    salario_medio NUMERIC,
    maior_salario NUMERIC,
    menor_salario NUMERIC,
    folha_total NUMERIC
)
LANGUAGE plpgsql
AS $$
DECLARE
    cur_dep CURSOR FOR
        SELECT id, nome FROM departamentos ORDER BY nome;

    v_dep_id INTEGER;
    v_dep_nome VARCHAR(100);

    v_total BIGINT;
    v_media NUMERIC;
    v_max NUMERIC;
    v_min NUMERIC;
    v_folha NUMERIC;
BEGIN
    OPEN cur_dep;

    LOOP
        FETCH cur_dep INTO v_dep_id, v_dep_nome;
        EXIT WHEN NOT FOUND;

        -- Calcular estatísticas para este departamento
        SELECT
            COUNT(*),
            ROUND(AVG(salario), 2),
            MAX(salario),
            MIN(salario),
            SUM(salario)
        INTO v_total, v_media, v_max, v_min, v_folha
        FROM funcionarios
        WHERE departamento_id = v_dep_id;

        -- Se o departamento tem funcionários, retorna a linha
        IF v_total > 0 THEN
            departamento := v_dep_nome;
            total_funcionarios := v_total;
            salario_medio := v_media;
            maior_salario := v_max;
            menor_salario := v_min;
            folha_total := v_folha;
            RETURN NEXT;
        ELSE
            -- Departamento sem funcionários
            departamento := v_dep_nome;
            total_funcionarios := 0;
            salario_medio := NULL;
            maior_salario := NULL;
            menor_salario := NULL;
            folha_total := 0;
            RETURN NEXT;
        END IF;
    END LOOP;

    CLOSE cur_dep;
END;
$$;

-- Executar
SELECT * FROM estatisticas_por_departamento();

4. Criar trigger que impede INSERT de produto com preço negativo

Crie uma tabela produtos com uma constraint CHECK de preço positivo. Adicionalmente, crie uma trigger function e um trigger BEFORE INSERT/UPDATE que valide se o preço é maior que zero. Se for menor ou igual a zero, lance uma exceção personalizada. A trigger também deve garantir que o nome do produto não seja vazio e normalizar o nome para maiúsculas.

Gabarito: 

-- Criar tabela
CREATE TABLE produtos (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    descricao TEXT,
    preco NUMERIC(10,2) NOT NULL CHECK (preco > 0),
    estoque INTEGER DEFAULT 0 CHECK (estoque >= 0),
    ativo BOOLEAN DEFAULT TRUE,
    criado_em TIMESTAMP DEFAULT now(),
    atualizado_em TIMESTAMP DEFAULT now()
);

-- Trigger function
CREATE OR REPLACE FUNCTION trg_validar_produto()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    -- Validar nome não vazio
    IF NEW.nome IS NULL OR TRIM(NEW.nome) = '' THEN
        RAISE EXCEPTION 'O nome do produto não pode ser vazio!';
    END IF;

    -- Validar preço positivo
    IF NEW.preco IS NULL OR NEW.preco <= 0 THEN
        RAISE EXCEPTION 'O preço do produto "%" deve ser maior que zero. Valor informado: %',
            NEW.nome, NEW.preco;
    END IF;

    -- Validar estoque não negativo
    IF NEW.estoque < 0 THEN
        RAISE EXCEPTION 'O estoque do produto "%" não pode ser negativo. Valor informado: %',
            NEW.nome, NEW.estoque;
    END IF;

    -- Normalizar nome para maiúsculas
    NEW.nome := UPPER(TRIM(NEW.nome));

    -- Atualizar timestamp
    NEW.atualizado_em := now();

    -- Se for INSERT, definir criado_em
    IF TG_OP = 'INSERT' THEN
        NEW.criado_em := COALESCE(NEW.criado_em, now());
    END IF;

    RETURN NEW;
END;
$$;

-- Criar trigger
CREATE TRIGGER trigger_validar_produto
    BEFORE INSERT OR UPDATE ON produtos
    FOR EACH ROW
    EXECUTE FUNCTION trg_validar_produto();
