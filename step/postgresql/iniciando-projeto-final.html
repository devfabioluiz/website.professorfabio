O Projeto Final

Neste projeto final você criará um sistema completo que integra banco relacional PostgreSQL, back-end Node.js e front-end React, formando uma aplicação full-stack pronta para produção.

O projeto será armazenado no repositório projeto-integrador-back-front-sql dentro de uma branch chamada projeto-final.

O sistema escolhido é o Sistema Escolar, com:

  • Banco Relacional — tabelas com chaves estrangeiras, JOINs e views para gerar boletins
  • API REST em Node.js — Express + pg (PostgreSQL) com autenticação JWT
  • Front-end React + Vite — consumindo a API e exibindo dados relacionados

🗄️ PASSO 1 — MODELAGEM DO BANCO (SISTEMA ESCOLAR)

Vamos modelar um Sistema Escolar com alunos, professores, disciplinas, turmas, matrículas e notas. Este modelo exige JOINs (para relacionar os dados) e Views (para facilitar consultas como o boletim).

Diagrama de Relacionamentos

Tabelas do Sistema

Tabela Descrição
alunos Dados dos alunos (nome, email, matrícula)
professores Dados dos professores (nome, email, especialidade)
disciplinas Disciplinas oferecidas (nome, carga horária)
turmas Turmas vinculadas a uma disciplina e professor (FK → disciplinas, professores)
matriculas Matrícula de alunos em turmas (FK → alunos, turmas)
notas Notas dos alunos por matrícula (FK → matriculas)

Um dos requisitos do sistema é exibir o boletim do aluno com: nome do aluno, disciplina, professor, nota e situação (aprovado/reprovado). Para isso, crie uma VIEW que centraliza os JOINs entre 5 tabelas de acordo com esses requisitos

const pool = require("../database/pg");

const listarTodos = async () => {
  const resultado = await pool.query(
    "SELECT * FROM alunos ORDER BY nome"
  );
  return resultado.rows;
};

const buscarPorId = async (id) => {
  const resultado = await pool.query(
    "SELECT * FROM alunos WHERE id = $1",
    [id]
  );
  return resultado.rows[0] || null;
};

const criar = async (dados) => {
  const { nome, email, matricula, data_nascimento } = dados;

  const resultado = await pool.query(
    `INSERT INTO alunos (nome, email, matricula, data_nascimento)
     VALUES ($1, $2, $3, $4)
     RETURNING *`,
    [nome, email, matricula, data_nascimento]
  );

  return resultado.rows[0];
};

const atualizar = async (id, dados) => {
  const { nome, email, data_nascimento } = dados;

  const resultado = await pool.query(
    `UPDATE alunos
     SET nome = $1, email = $2, data_nascimento = $3
     WHERE id = $4
     RETURNING *`,
    [nome, email, data_nascimento, id]
  );

  return resultado.rows[0] || null;
};

const deletar = async (id) => {
  const resultado = await pool.query(
    "DELETE FROM alunos WHERE id = $1 RETURNING *",
    [id]
  );

  return resultado.rows[0] || null;
};

module.exports = {
  listarTodos,
  buscarPorId,
  criar,
  atualizar,
  deletar,
};

No terminal coloque o código abaixo:

const pool = require("../database/pg");

const boletimDoAluno = async (alunoId) => {
  const resultado = await pool.query(
    "SELECT * FROM vw_boletim WHERE aluno_id = $1 ORDER BY disciplina_nome, data_avaliacao",
    [alunoId]
  );

  return resultado.rows;
};

const boletimDaTurma = async (turmaId) => {
  const resultado = await pool.query(
    `SELECT * FROM vw_boletim
     WHERE turma_codigo = (SELECT codigo FROM turmas WHERE id = $1)
     ORDER BY aluno_nome, disciplina_nome`,
    [turmaId]
  );

  return resultado.rows;
};

const mediaPorDisciplina = async (alunoId) => {
  const resultado = await pool.query(
    `SELECT
       disciplina_nome,
       ROUND(AVG(nota), 2) AS media,
       COUNT(*) AS total_avaliacoes
     FROM vw_boletim
     WHERE aluno_id = $1 AND nota IS NOT NULL
     GROUP BY disciplina_nome
     ORDER BY disciplina_nome`,
    [alunoId]
  );

  return resultado.rows;
};

module.exports = {
  boletimDoAluno,
  boletimDaTurma,
  mediaPorDisciplina,
};

🤓 Explicando o BoletimRepository

🔹 Consultando a VIEW

O repositório consulta a VIEW vw_boletim como se fosse uma tabela comum. Toda a complexidade dos JOINs entre 5 tabelas fica encapsulada na VIEW.

🔹 Funções do repositório

  • boletimDoAluno — retorna todas as notas do aluno com detalhes das disciplinas
  • boletimDaTurma — retorna o boletim de todos os alunos de uma turma
  • mediaPorDisciplina — calcula a média do aluno por disciplina usando GROUP BY e ROUND(AVG())

🔹 Por que a VIEW é importante aqui?

Sem a VIEW, cada uma dessas funções precisaria repetir os 5 JOINs. Com a VIEW, o código fica limpo e a lógica fica centralizada no banco de dados — facilitando manutenção e garantindo consistência.

Passo 2.8 — Middlewares de Autenticação

Crie src/middlewares/autenticar.js: 

const jwt = require("jsonwebtoken");

const JWT_SECRET = process.env.JWT_SECRET || "segredo";

const autenticar = (req, res, next) => {
  const authHeader = req.headers.authorization;

  if (!authHeader) {
    return res.status(401).json({ erro: "Token não informado" });
  }

  const token = authHeader.split(" ")[1];

  try {
    const payload = jwt.verify(token, JWT_SECRET);
    req.usuarioId = payload.id;
    req.usuarioEmail = payload.email;
    req.usuarioRole = payload.role;
    next();
  } catch (erro) {
    return res.status(401).json({ erro: "Token inválido ou expirado" });
  }
};

module.exports = autenticar;

Crie src/middlewares/admin.js: 

const admin = (req, res, next) => {

Passo 2.9 — Controller de Autenticação

Crie src/controllers/authController.js (registro e login de professores/administradores): 

const bcrypt = require("bcryptjs");
const jwt = require("jsonwebtoken");
const pool = require("../database/pg");

const JWT_SECRET = process.env.JWT_SECRET || "segredo";

const registrar = async (req, res) => {
  const { nome, email, senha, role } = req.body;

  const existe = await pool.query(
    "SELECT * FROM professores WHERE email = $1",
    [email]
  );

  if (existe.rows.length > 0) {
    return res.status(400).json({ erro: "Email já cadastrado" });
  }

  const senhaHash = await bcrypt.hash(senha, 10);

  const resultado = await pool.query(
    `INSERT INTO professores (nome, email, senha, role)
     VALUES ($1, $2, $3, $4)
     RETURNING id, nome, email, role`,
    [nome, email, senhaHash, role || "admin"]
  );

  const professor = resultado.rows[0];

  const token = jwt.sign(
    { id: professor.id, email: professor.email, role: professor.role },
    JWT_SECRET,
    { expiresIn: "7d" }
  );

  res.status(201).json({
    mensagem: "Professor registrado com sucesso",
    token,
    usuario: professor,
  });
};

const login = async (req, res) => {
  const { email, senha } = req.body;

  const resultado = await pool.query(
    "SELECT * FROM professores WHERE email = $1",
    [email]
  );

  const professor = resultado.rows[0];

  if (!professor) {
    return res
      .status(401)
      .json({ erro: "Email ou senha inválidos" });
  }

  const senhaValida = await bcrypt.compare(senha, professor.senha);

  if (!senhaValida) {
    return res
      .status(401)
      .json({ erro: "Email ou senha inválidos" });
  }

  const token = jwt.sign(
    {
      id: professor.id,
      email: professor.email,
      role: professor.role,
    },
    JWT_SECRET,
    { expiresIn: "7d" }
  );

  res.json({
    mensagem: "Login realizado com sucesso",
    token,
    usuario: {
      id: professor.id,
      nome: professor.nome,
      email: professor.email,
      role: professor.role,
    },
  });
};

const perfil = async (req, res) => {
  const resultado = await pool.query(
    "SELECT id, nome, email, role FROM professores WHERE id = $1",
    [req.usuarioId]
  );

  const professor = resultado.rows[0];

  if (!professor) {
    return res
      .status(404)
      .json({ erro: "Professor não encontrado" });
  }

  res.json({ usuario: professor });
};

module.exports = { registrar, login, perfil };

Passo 2.11 — Criar o app.js

const express = require("express");
const helmet = require("helmet");
const cors = require("cors");
const compression = require("compression");
const rateLimit = require("express-rate-limit");

const alunoRoutes = require("./src/routes/alunoRoutes");
const authRoutes = require("./src/routes/authRoutes");
const boletimRoutes = require("./src/routes/boletimRoutes");

const app = express();

app.use(helmet());
app.use(cors({ origin: process.env.CORS_ORIGIN || "*" }));
app.use(compression());

const limitador = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 100,
  message: { erro: "Muitas requisições. Tente novamente mais tarde." },
});
app.use("/api/", limitador);

app.use(express.json());

app.get("/", (req, res) => {
  res.json({
    mensagem: "API do Sistema Escolar funcionando!",
    versao: "1.0.0",
    endpoints: {
      alunos: "GET /alunos, POST /alunos, PUT /alunos/:id, DELETE /alunos/:id",
      auth: "POST /registrar, POST /login, GET /me",
      boletim: "GET /boletim/:alunoId, GET /boletim/:alunoId/media",
    },
  });
});

app.use(alunoRoutes);
app.use(authRoutes);
app.use(boletimRoutes);

app.use((erro, req, res, next) => {
  console.error("Erro interno:", erro.message);
  res.status(erro.status || 500).json({
    erro:
      process.env.NODE_ENV === "production"
        ? "Erro interno do servidor"
        : erro.message,
  });
});

module.exports = app;

🤓 Por que o proxy?

Sem o proxy, o navegador bloquearia as requisições do React (porta 5173) para o Node.js (porta 3000) por causa da política de CORS. O proxy do Vite faz o papel de intermediário: o React chama /alunos (mesma origem) e o Vite redireciona para http://localhost:3000/alunos.

Passo 3.3 — Criar o serviço de API

Crie src/services/api.js para centralizar as chamadas HTTP: 

const API_BASE = "";

async function request(endpoint, options = {}) {
  const token = localStorage.getItem("token");

  const headers = {
    "Content-Type": "application/json",
    ...(token ? { Authorization: `Bearer ${token}` } : {}),
    ...options.headers,
  };

  const response = await fetch(`${API_BASE}${endpoint}`, {
    ...options,
    headers,
  });

  const data = await response.json();

  if (!response.ok) {
    throw new Error(data.erro || "Erro na requisição");
  }

  return data;
}

export const api = {
  login: (email, senha) =>
    request("/login", {
      method: "POST",
      body: JSON.stringify({ email, senha }),
    }),

  registrar: (nome, email, senha) =>
    request("/registrar", {
      method: "POST",
      body: JSON.stringify({ nome, email, senha }),
    }),

  listarAlunos: () => request("/alunos"),

  criarAluno: (dados) =>
    request("/alunos", {
      method: "POST",
      body: JSON.stringify(dados),
    }),

  buscarBoletim: (alunoId) => request(`/boletim/${alunoId}`),

  buscarMedia: (alunoId) => request(`/boletim/${alunoId}/media`),

  perfil: () => request("/me"),
};
