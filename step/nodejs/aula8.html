Cliente Web no Node.js

Até agora usamos o Node.js como servidor. Mas ele também pode atuar como cliente, fazendo requisições HTTP para outros servidores, APIs externas e serviços web.

Isso é útil para:

  • Consumir APIs de terceiros (ViaCEP, OpenWeather, GitHub, etc.)
  • Integrar microsserviços
  • Buscar dados de fontes externas
  • Web scraping (coleta de dados de sites)
  • Enviar notificações para serviços externos

Fazendo Requisições HTTP

O Node.js oferece várias formas de fazer requisições HTTP como cliente:

Método Vantagem Desvantagem
http.get() Nativo, sem dependências Verboso, não suporta Promises nativamente
fetch() Nativo (Node 18+), simples, Promise-based Menos controle de baixo nível
axios Rico em recursos, interceptors Dependência externa

Usando o módulo http nativo (cliente)

const http = require('http');

const options = {
  hostname: 'jsonplaceholder.typicode.com',
  path: '/posts/1',
  method: 'GET'
};

const req = http.request(options, (res) => {
  let dados = '';

  res.on('data', (pedaco) => {
    dados += pedaco;
  });

  res.on('end', () => {
    const post = JSON.parse(dados);
    console.log('Título:', post.title);
    console.log('Status:', res.statusCode);
  });
});

req.on('error', (erro) => {
  console.log('Erro:', erro.message);
});

req.end();

Note que o padrão é o mesmo de streams: os dados chegam em pedaços no evento data e o evento end indica conclusão.

Enviando dados com POST

const http = require('http');

const dados = JSON.stringify({
  title: 'Novo Post via Node.js',
  body: 'Corpo do post',
  userId: 1
});

const options = {
  hostname: 'jsonplaceholder.typicode.com',
  path: '/posts',
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Content-Length': Buffer.byteLength(dados)
  }
};

const req = http.request(options, (res) => {
  let resposta = '';

  res.on('data', (pedaco) => {
    resposta += pedaco;
  });

  res.on('end', () => {
    console.log('Resposta:', JSON.parse(resposta));
  });
});

req.write(dados);
req.end();

Usando fetch (nativo a partir do Node 18)

async function buscarPost() {
  const resposta = await fetch(
    'https://jsonplaceholder.typicode.com/posts/1'
  );

  if (!resposta.ok) {
    throw new Error(`Erro HTTP: ${resposta.status}`);
  }

  const post = await resposta.json();

  console.log('Título:', post.title);
  console.log('Corpo:', post.body);
}

buscarPost();

Fetch com POST

async function criarPost() {
  const resposta = await fetch(
    'https://jsonplaceholder.typicode.com/posts',
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        title: 'Post criado com fetch',
        body: 'Corpo do post',
        userId: 1
      })
    }
  );

  const post = await resposta.json();
  console.log('Criado:', post);
}

criarPost();

async function buscarEndereco(cep) {
  const cepLimpo = cep.replace(/\D/g, '');

  const resposta = await fetch(
    `https://viacep.com.br/ws/${cepLimpo}/json/`

  if (dados.erro) {

O que são Formulários no Backend?

Quando um navegador envia um formulário, os dados podem chegar em diferentes formatos. O Express precisa entender esses formatos para conseguir acessar req.body.

Formatos de envio

Content-Type Origem Middleware Express
application/json APIs, fetch, axios express.json()
application/x-www-form-urlencoded Formulários HTML tradicionais express.urlencoded()
multipart/form-data Upload de arquivos multer

Formulário HTML tradicional

<form action="/cadastrar" method="POST">
  <input type="text" name="nome" placeholder="Nome" />
  <input type="email" name="email" placeholder="Email" />
  <input type="password" name="senha" placeholder="Senha" />
  <button type="submit">Cadastrar</button>
</form>

Esse formulário envia os dados no formato application/x-www-form-urlencoded: 

nome=Maria&email=maria@email.com&senha=123456

Configurando o Express para formulários HTML

const express = require('express');
const app = express();

app.use(express.json());

app.use(express.urlencoded({ extended: true }));

app.post('/cadastrar', (req, res) => {
  console.log(req.body);

  res.send('Cadastro recebido!');
});

🤓 Explicando o código 📝

🔹 express.urlencoded({ extended: true })

Esse middleware interpreta dados enviados por formulários HTML tradicionais. A opção extended: true permite objetos aninhados (ex: endereco[rua]=X). Com extended: false, apenas strings e arrays simples.

🔹 req.body

Após registrar os middlewares express.json() e express.urlencoded(), o req.body estará disponível automaticamente para requisições POST, PUT e PATCH.

O que são Cookies?

Cookies são pequenos pedaços de dados que o servidor envia para o navegador e que o navegador devolve automaticamente em requisições seguintes. Eles são armazenados no lado do cliente.

Usos comuns de cookies:

  • Manter preferências do usuário (tema, idioma)
  • Rastrear sessões de navegação
  • Armazenar tokens de autenticação
  • Carregar carrinhos de compras

Instalando o cookie-parser

npm install cookie-parser

Enviando e lendo cookies

const express = require('express');
const cookieParser = require('cookie-parser');

const app = express();

app.use(cookieParser());

app.get('/definir-cookie', (req, res) => {
  res.cookie('tema', 'escuro', {
    maxAge: 24 * 60 * 60 * 1000,
    httpOnly: true
  });

  res.send('Cookie definido!');
});

app.get('/ler-cookie', (req, res) => {
  const tema = req.cookies.tema || 'claro';

  res.json({ tema: tema });
});

app.get('/remover-cookie', (req, res) => {
  res.clearCookie('tema');

  res.send('Cookie removido!');
});

🤓 Explicando o código 📝

🔹 res.cookie()

Define um cookie no navegador do cliente. Principais opções:

Opção Descrição
maxAge Tempo de vida em milissegundos
httpOnly Impede acesso via JavaScript no navegador (mais seguro)
secure Só envia em conexões HTTPS
sameSite Proteção contra CSRF (strict, lax, none)

🔹 req.cookies

Contém todos os cookies enviados pelo navegador na requisição atual. É um objeto onde a chave é o nome do cookie.

O que são Sessões?

Sessões permitem manter dados do usuário entre múltiplas requisições. Diferente de cookies (armazenados no cliente), os dados da sessão ficam no servidor. Apenas um identificador (session ID) é armazenado no cookie do navegador.

Instalando express-session

npm install express-session

Configurando sessões

const express = require('express');
const session = require('express-session');

const app = express();

app.use(session({
  secret: 'segredo-muito-forte-aqui',
  resave: false,
  saveUninitialized: false,
  cookie: {
    maxAge: 30 * 60 * 1000,
    httpOnly: true
  }
}));

app.post('/login', (req, res) => {
  const { email } = req.body;

  req.session.usuario = {
    email: email,
    logadoEm: new Date()
  };

  res.json({ mensagem: 'Login realizado' });
});

app.get('/perfil', (req, res) => {
  if (!req.session.usuario) {
    return res.status(401).json({
      erro: 'Faça login primeiro'
    });
  }

  res.json({
    usuario: req.session.usuario
  });
});

app.post('/logout', (req, res) => {
  req.session.destroy((erro) => {
    if (erro) {
      return res.status(500).json({ erro: 'Erro ao encerrar sessão' });
    }

    res.json({ mensagem: 'Logout realizado' });
  });
});

🤓 Explicando o código 📝

🔹 secret

Usado para assinar o cookie de sessão, garantindo que não foi alterado pelo cliente. Deve ser uma string longa e aleatória — NUNCA coloque em código (use variável de ambiente).

🔹 resave: false

Evita salvar a sessão de volta no armazenamento se ela não foi modificada. Reduz operações desnecessárias.

🔹 saveUninitialized: false

Não cria uma sessão para usuários que ainda não "fizeram nada". Importante para compliance com leis como LGPD/GDPR.

🔹 req.session.destroy()

Remove completamente a sessão do servidor e o cookie do navegador.

🔹 Sessão vs JWT

Sessão (Stateful) JWT (Stateless)
Dados ficam no servidor Dados ficam no token
Fácil invalidar (destroy) Difícil invalidar (precisa blacklist)
Não escala bem horizontalmente Escala melhor (qualquer servidor valida)
Ideal para apps tradicionais Ideal para APIs e microsserviços

const logger = require("../utils/logger");

const buscarCEP = async (req, res) => {
  const { cep } = req.params;

  const cepLimpo = cep.replace(/\D/g, "");

  if (cepLimpo.length !== 8) {
    return res.status(400).json({
      erro: "CEP deve ter 8 dígitos",
    });
  }

  try {
    const resposta = await fetch(
      `https://viacep.com.br/ws/${cepLimpo}/json/`
    );

    if (!resposta.ok) {
      throw new Error("Serviço de CEP indisponível");
    }

    const dados = await resposta.json();

    if (dados.erro) {
      return res.status(404).json({
        erro: "CEP não encontrado",
      });
    }

    logger.info(`CEP consultado: ${cepLimpo}`);

    res.json({
      cep: dados.cep,
      logradouro: dados.logradouro,
      bairro: dados.bairro,
      cidade: dados.localidade,
      estado: dados.uf,
    });
  } catch (erro) {
    logger.erro(`Erro ao buscar CEP: ${erro.message}`);

    res.status(500).json({
      erro: "Erro ao consultar serviço de CEP",
    });
  }
};

module.exports = {
  buscarCEP,
};

const logger = require("../utils/logger");

const definirTema = (req, res) => {
  const { tema } = req.body;

  if (!tema || !["claro", "escuro"].includes(tema)) {
    return res.status(400).json({
      erro: "Tema inválido. Use 'claro' ou 'escuro'",
    });
  }

  res.cookie("temaPreferido", tema, {
    maxAge: 30 * 24 * 60 * 60 * 1000,
    httpOnly: true,
  });

  logger.info(`Tema alterado para: ${tema}`);

  res.json({ tema: tema });
};

const obterTema = (req, res) => {
  const tema = req.cookies.temaPreferido || "claro";

  res.json({ tema: tema });
};

const abrirSessao = (req, res) => {
  req.session.visitas = (req.session.visitas || 0) + 1;

  res.json({
    mensagem: "Sessão iniciada",
    visitas: req.session.visitas,
    sessaoId: req.sessionID,
  });
};

const statusSessao = (req, res) => {
  if (!req.session.visitas) {
    return res.json({
      autenticado: false,
      visitas: 0,
    });
  }

  res.json({
    autenticado: true,
    visitas: req.session.visitas,
    sessaoId: req.sessionID,
  });
};

const encerrarSessao = (req, res) => {
  req.session.destroy((erro) => {
    if (erro) {
      return res.status(500).json({
        erro: "Erro ao encerrar sessão",
      });
    }

    res.clearCookie("connect.sid");

    res.json({ mensagem: "Sessão encerrada" });
  });
};

module.exports = {
  definirTema,
  obterTema,
  abrirSessao,
  statusSessao,
  encerrarSessao,
};
