1. Por que ViewSets e Routers?

Na Aula 4, para criar um CRUD de usuários precisamos escrever cada endpoint manualmente: 


# Aula 4 — forma manual (verbosa)
@api_view(['GET'])
def listar_usuarios(request):
    usuarios = Usuario.objects.all()
    serializer = UsuarioSerializer(usuarios, many=True)
    return Response(serializer.data)

@api_view(['POST'])
def criar_usuario(request):
    ...

@api_view(['PUT'])
def atualizar_usuario(request, id):
    ...

@api_view(['DELETE'])
def deletar_usuario(request, id):
    ...

# E no urls.py:
urlpatterns = [
    path('api/usuarios/', listar_usuarios),
    path('api/usuarios/criar/', criar_usuario),
    path('api/usuarios//', atualizar_usuario),
    path('api/usuarios//deletar/', deletar_usuario),
]

Com ViewSets + Routers, o mesmo CRUD se resume a: 


# ViewSet — define a lógica
class UsuarioViewSet(viewsets.ModelViewSet):
    queryset = Usuario.objects.all()
    serializer_class = UsuarioSerializer

# Router — gera automaticamente:
# GET    /usuarios/         → list
# POST   /usuarios/         → create
# GET    /usuarios/{id}/    → retrieve
# PUT    /usuarios/{id}/    → update
# PATCH  /usuarios/{id}/    → partial_update
# DELETE /usuarios/{id}/    → destroy

Economia: de ~40 linhas para ~6 linhas. E ainda ganha endpoints padronizados seguindo convenções REST.

2. ModelViewSet — O canivete suíço do DRF

ModelViewSet combina todos os mixins de CRUD em uma única classe. Ele herda de:

Classe Base Método HTTP Ação Endpoint
ListModelMixin GET list() /tarefas/
CreateModelMixin POST create() /tarefas/
RetrieveModelMixin GET retrieve() /tarefas/{id}/
UpdateModelMixin PUT/PATCH update() /tarefas/{id}/
DestroyModelMixin DELETE destroy() /tarefas/{id}/

Você pode sobrescrever qualquer método para adicionar lógica customizada. Também pode criar ações extras com @action: 


from rest_framework.decorators import action
from rest_framework.response import Response

class TarefaViewSet(viewsets.ModelViewSet):
    queryset = Tarefa.objects.all()
    serializer_class = TarefaSerializer

    # Ação extra: POST /tarefas/{id}/concluir/
    @action(detail=True, methods=['post'])
    def concluir(self, request, pk=None):
        tarefa = self.get_object()
        tarefa.concluida = True
        tarefa.save()
        return Response({'status': 'tarefa concluída'})

    # Ação extra: GET /tarefas/concluidas/
    @action(detail=False, methods=['get'])
    def concluidas(self, request):
        concluidas = self.queryset.filter(concluida=True)
        serializer = self.get_serializer(concluidas, many=True)
        return Response(serializer.data)

detail=True = ação sobre um item específico (recebe pk). detail=False = ação sobre a coleção.

3. Paginação

Quando sua API tem centenas ou milhares de registros, você não quer retornar tudo de uma vez. A paginação divide os resultados em páginas.

3.1 — PageNumberPagination (mais comum)

Divide por páginas numeradas. O cliente pede ?page=2.


# settings.py
REST_FRAMEWORK = {
    'DEFAULT_PAGINATION_CLASS': 'rest_framework.pagination.PageNumberPagination',
    'PAGE_SIZE': 10,  # 10 itens por página
}

Resposta da API (com paginação):


{
    "count": 85,
    "next": "http://localhost:8000/api/tarefas/?page=3",
    "previous": "http://localhost:8000/api/tarefas/?page=1",
    "results": [
        { "id": 11, "titulo": "Tarefa 11", ... },
        { "id": 12, "titulo": "Tarefa 12", ... }
    ]
}

3.2 — LimitOffsetPagination

O cliente controla quantos itens quer e a partir de qual posição: ?limit=20&offset=40.


# settings.py (alternativa)
REST_FRAMEWORK = {
    'DEFAULT_PAGINATION_CLASS': 'rest_framework.pagination.LimitOffsetPagination',
    'PAGE_SIZE': 25,
}

3.3 — Paginação customizada por ViewSet

Se você quer tamanhos diferentes por endpoint:


from rest_framework.pagination import PageNumberPagination

class TarefaPaginacao(PageNumberPagination):
    page_size = 5
    page_size_query_param = 'tamanho'    # cliente pode mudar: ?tamanho=20
    max_page_size = 100                   # limite máximo

class TarefaViewSet(viewsets.ModelViewSet):
    queryset = Tarefa.objects.all()
    serializer_class = TarefaSerializer
    pagination_class = TarefaPaginacao

4. Permissões no DRF

O DRF tem um sistema de permissões que controla quem pode acessar cada endpoint. Diferente do @login_required do Django tradicional (que redireciona para página de login), as permissões do DRF retornam 401 Unauthorized ou 403 Forbidden — apropriado para APIs.

Permissão Comportamento
AllowAny Qualquer um pode acessar (inclusive não autenticados)
IsAuthenticated Apenas usuários logados (retorna 401 se não autenticado)
IsAdminUser Apenas superusuários (is_staff=True)
IsAuthenticatedOrReadOnly Leitura pública, escrita só para autenticados

Configurando permissões globais vs por ViewSet


# settings.py — padrão global (opcional)
REST_FRAMEWORK = {
    'DEFAULT_PERMISSION_CLASSES': [
        'rest_framework.permissions.IsAuthenticated',
    ],
}

# Por ViewSet — sobrescreve o global
from rest_framework.permissions import IsAuthenticated, AllowAny

class TarefaViewSet(viewsets.ModelViewSet):
    queryset = Tarefa.objects.all()
    serializer_class = TarefaSerializer
    permission_classes = [IsAuthenticated]  # só logados

class TarefaPublicaViewSet(viewsets.ReadOnlyModelViewSet):
    queryset = Tarefa.objects.filter(publica=True)
    serializer_class = TarefaSerializer
    permission_classes = [AllowAny]  # qualquer um pode ver

Permissão customizada: dono do recurso


from rest_framework.permissions import BasePermission

class EhDono(BasePermission):
    """Permite acesso apenas ao dono da tarefa"""
    def has_object_permission(self, request, view, obj):
        return obj.usuario == request.user

class TarefaViewSet(viewsets.ModelViewSet):
    permission_classes = [IsAuthenticated, EhDono]

    def perform_create(self, serializer):
        # Associa a tarefa ao usuário logado automaticamente
        serializer.save(usuario=self.request.user)

5. Filtros e Busca

O DRF integra com django-filter para filtros avançados e oferece backends nativos de busca e ordenação.

5.1 — SearchFilter (busca textual)


from rest_framework.filters import SearchFilter

class TarefaViewSet(viewsets.ModelViewSet):
    queryset = Tarefa.objects.all()
    serializer_class = TarefaSerializer
    filter_backends = [SearchFilter]
    search_fields = ['titulo', 'descricao']

# Uso: GET /tarefas/?search=comprar

5.2 — OrderingFilter (ordenação)


from rest_framework.filters import SearchFilter, OrderingFilter

class TarefaViewSet(viewsets.ModelViewSet):
    filter_backends = [SearchFilter, OrderingFilter]
    search_fields = ['titulo']
    ordering_fields = ['data_criacao', 'prioridade', 'titulo']
    ordering = ['-data_criacao']  # ordenação padrão

# Uso: GET /tarefas/?ordering=prioridade,-data_criacao

Arquivo: tarefas/models.py


from django.db import models
from django.contrib.auth.models import User

class Tarefa(models.Model):
    PRIORIDADE_CHOICES = [
        ('baixa', 'Baixa'),
        ('media', 'Média'),
        ('alta', 'Alta'),
    ]

    titulo = models.CharField(max_length=200)
    descricao = models.TextField(blank=True, null=True)
    concluida = models.BooleanField(default=False)
    prioridade = models.CharField(max_length=10, choices=PRIORIDADE_CHOICES, default='media')
    data_criacao = models.DateTimeField(auto_now_add=True)
    data_conclusao = models.DateTimeField(null=True, blank=True)
    usuario = models.ForeignKey(User, on_delete=models.CASCADE, related_name='tarefas')

    class Meta:
        ordering = ['-data_criacao']

    def __str__(self):
        return self.titulo

from rest_framework import serializers
from .models import Tarefa

class TarefaSerializer(serializers.ModelSerializer):
    usuario_nome = serializers.ReadOnlyField(source='usuario.username')

    class Meta:
        model = Tarefa
        fields = [
            'id', 'titulo', 'descricao', 'concluida', 'prioridade',
            'data_criacao', 'data_conclusao', 'usuario', 'usuario_nome'
        ]
        read_only_fields = ['usuario', 'data_criacao', 'data_conclusao']

    def validate_titulo(self, value):
        if len(value) < 3:
            raise serializers.ValidationError("O título deve ter pelo menos 3 caracteres.")
        return value

Passo 6 — Criar o ViewSet

Arquivo: tarefas/views.py


from rest_framework import viewsets, status
from rest_framework.decorators import action
from rest_framework.response import Response
from rest_framework.permissions import IsAuthenticated
from rest_framework.filters import SearchFilter, OrderingFilter
from django_filters.rest_framework import DjangoFilterBackend
from django.utils import timezone

from .models import Tarefa
from .serializers import TarefaSerializer
from .permissions import EhDono

class TarefaViewSet(viewsets.ModelViewSet):
    queryset = Tarefa.objects.all()
    serializer_class = TarefaSerializer
    permission_classes = [IsAuthenticated, EhDono]
    filter_backends = [DjangoFilterBackend, SearchFilter, OrderingFilter]
    filterset_fields = ['concluida', 'prioridade']
    search_fields = ['titulo', 'descricao']
    ordering_fields = ['data_criacao', 'prioridade', 'titulo']
    ordering = ['-data_criacao']

    def get_queryset(self):
        # Cada usuário só vê as próprias tarefas
        return Tarefa.objects.filter(usuario=self.request.user)

    def perform_create(self, serializer):
        # Associa a tarefa ao usuário logado automaticamente
        serializer.save(usuario=self.request.user)

    # Ação extra: POST /tarefas/{id}/concluir/
    @action(detail=True, methods=['post'])
    def concluir(self, request, pk=None):
        tarefa = self.get_object()
        if tarefa.concluida:
            return Response(
                {'erro': 'Esta tarefa já está concluída.'},
                status=status.HTTP_400_BAD_REQUEST
            )
        tarefa.concluida = True
        tarefa.data_conclusao = timezone.now()
        tarefa.save()
        return Response({'status': 'tarefa concluída com sucesso'})

    # Ação extra: POST /tarefas/{id}/reabrir/
    @action(detail=True, methods=['post'])
    def reabrir(self, request, pk=None):
        tarefa = self.get_object()
        if not tarefa.concluida:
            return Response(
                {'erro': 'Esta tarefa não está concluída.'},
                status=status.HTTP_400_BAD_REQUEST
            )
        tarefa.concluida = False
        tarefa.data_conclusao = None
        tarefa.save()
        return Response({'status': 'tarefa reaberta com sucesso'})

    # Ação extra: GET /tarefas/estatisticas/
    @action(detail=False, methods=['get'])
    def estatisticas(self, request):
        tarefas = self.get_queryset()
        total = tarefas.count()
        concluidas = tarefas.filter(concluida=True).count()
        pendentes = total - concluidas

        return Response({
            'total': total,
            'concluidas': concluidas,
            'pendentes': pendentes,
            'porcentagem_concluida': round((concluidas / total * 100) if total > 0 else 0, 1),
        })

Dever de Casa

1. Adicione um endpoint de filtro por data

Crie uma ação extra no ViewSet que retorna as tarefas criadas nos últimos N dias. Exemplo: GET /api/tarefas/recentes/?dias=7.

Gabarito: 

from django.utils import timezone
from datetime import timedelta

class TarefaViewSet(viewsets.ModelViewSet):
    # ... código existente ...

    @action(detail=False, methods=['get'])
    def recentes(self, request):
        dias = int(request.query_params.get('dias', 7))
        data_limite = timezone.now() - timedelta(days=dias)
        tarefas = self.get_queryset().filter(data_criacao__gte=data_limite)
        page = self.paginate_queryset(tarefas)

        if page is not None:
            serializer = self.get_serializer(page, many=True)
            return self.get_paginated_response(serializer.data)

        serializer = self.get_serializer(tarefas, many=True)
        return Response(serializer.data)
