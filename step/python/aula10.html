1. Classes e Objetos

Uma classe é um molde. Um objeto é uma instância concreta desse molde. Pense numa fôrma de bolo (classe) e nos bolos que saem dela (objetos).

Estrutura básica


class Livro:
    # Atributo de classe (compartilhado por todos os objetos)
    total_livros = 0

    def __init__(self, titulo, autor, paginas):
        # Atributos de instância (cada objeto tem os seus)
        self.titulo = titulo
        self.autor = autor
        self.paginas = paginas
        self.emprestado = False

        # Incrementa o contador global
        Livro.total_livros += 1

    def emprestar(self):
        if not self.emprestado:
            self.emprestado = True
            return f"'{self.titulo}' foi emprestado."
        return f"'{self.titulo}' já está emprestado."

    def devolver(self):
        if self.emprestado:
            self.emprestado = False
            return f"'{self.titulo}' foi devolvido."
        return f"'{self.titulo}' não estava emprestado."


# Criando objetos (instâncias)
livro1 = Livro("Dom Casmurro", "Machado de Assis", 256)
livro2 = Livro("1984", "George Orwell", 328)

print(livro1.titulo)          # Dom Casmurro
print(livro1.emprestar())     # 'Dom Casmurro' foi emprestado.
print(livro1.emprestar())     # 'Dom Casmurro' já está emprestado.
print(livro1.devolver())      # 'Dom Casmurro' foi devolvido.
print(Livro.total_livros)     # 2

self — o que é?

self representa a própria instância. Quando você chama livro1.emprestar(), o Python traduz para Livro.emprestar(livro1). É assim que cada objeto sabe quais são seus próprios dados.

Atributo de classe vs Atributo de instância

Tipo Definido com Compartilhado? Exemplo
De classe Fora do __init__ Sim (todos os objetos) total_livros
De instância Dentro do __init__ com self. Não (cada um tem o seu) titulo

2. Encapsulamento

Encapsulamento é esconder detalhes internos do objeto e expor apenas o que é necessário. Em Python, usamos convenções de nomenclatura.

Convenção Significado Exemplo
atributo Público — qualquer um pode acessar self.titulo
_atributo Protegido — "não mexa, mas se quiser..." self._estoque
__atributo Privado — Python renomeia (name mangling) self.__senha

@property — getters e setters elegantes


class Usuario:
    def __init__(self, nome, email):
        self.nome = nome
        self._email = None       # atributo interno
        self.email = email        # chama o setter!

    @property
    def email(self):
        """Getter — acessado como usuario.email"""
        return self._email

    @email.setter
    def email(self, novo_email):
        """Setter — usuario.email = 'novo@email.com'"""
        if '@' not in novo_email:
            raise ValueError("Email inválido! Deve conter @")
        self._email = novo_email.lower()

    @property
    def dominio(self):
        """Propriedade calculada — só getter, sem setter"""
        return self._email.split('@')[1] if self._email else None


# Testando
usuario = Usuario("Ana", "ana@email.com")
print(usuario.email)      # ana@email.com
print(usuario.dominio)    # email.com
usuario.email = "ANA@NOVA.COM"
print(usuario.email)      # ana@nova.com

# usuario.email = "email_invalido"  # ValueError!

3. Duck Typing

Em Python, não importa o tipo do objeto — importa o que ele faz. Se um objeto tem o método quack(), ele pode ser tratado como pato. 


# Duck Typing: a função não exige um tipo específico
def imprimir_dados(objeto):
    """Qualquer objeto que tenha .titulo e .autor funciona"""
    print(f"'{objeto.titulo}' por {objeto.autor}")

class Livro:
    def __init__(self, titulo, autor):
        self.titulo = titulo
        self.autor = autor

class Filme:
    def __init__(self, titulo, autor):
        self.titulo = titulo
        self.autor = autor  # diretor

class Musica:
    def __init__(self, titulo, autor):
        self.titulo = titulo
        self.autor = autor  # artista

# Todos funcionam, mesmo sendo de classes diferentes!
imprimir_dados(Livro("Dom Casmurro", "Machado de Assis"))
imprimir_dados(Filme("O Poderoso Chefão", "Francis Ford Coppola"))
imprimir_dados(Musica("Bohemian Rhapsody", "Queen"))

4. Métodos Mágicos (Dunder Methods)

Métodos com __nome__ são chamados automaticamente pelo Python em situações específicas. Eles tornam seus objetos "nativos".

Os mais importantes

Método Quando é chamado Gatilho
__init__ Ao criar o objeto obj = Classe()
__str__ Representação para o usuário print(obj), str(obj)
__repr__ Representação para o desenvolvedor repr(obj), console interativo
__eq__ Comparação de igualdade obj1 == obj2
__len__ Tamanho do objeto len(obj)
__iter__ Iteração em loop for item in obj:
__getitem__ Acesso por índice obj[0]
__contains__ Verificação de pertencimento item in obj

Exemplo: classe Acervo com métodos mágicos


class Acervo:
    def __init__(self, nome):
        self.nome = nome
        self._itens = []

    def adicionar(self, item):
        self._itens.append(item)

    # Métodos mágicos
    def __str__(self):
        return f"📚 Acervo '{self.nome}' — {len(self._itens)} itens"

    def __repr__(self):
        return f"Acervo(nome='{self.nome}', itens={len(self._itens)})"

    def __len__(self):
        return len(self._itens)

    def __getitem__(self, index):
        return self._itens[index]

    def __iter__(self):
        return iter(self._itens)

    def __contains__(self, item):
        return item in self._itens

    def __eq__(self, outro):
        if not isinstance(outro, Acervo):
            return False
        return self.nome == outro.nome and len(self) == len(outro)


# Usando os métodos mágicos
acervo = Acervo("Ficção Científica")
acervo.adicionar("Duna")
acervo.adicionar("Neuromancer")
acervo.adicionar("Fundação")

print(acervo)                 # 📚 Acervo 'Ficção Científica' — 3 itens
print(len(acervo))            # 3
print(acervo[0])              # Duna

for item in acervo:           # Iteração graças ao __iter__
    print(f"  → {item}")

print("Duna" in acervo)       # True (graças ao __contains__)
print("Harry Potter" in acervo)  # False

5. Herança

Herança permite que uma classe filha herde atributos e métodos de uma classe pai, reutilizando e estendendo código.

5.1 — Herança Simples


class ItemBiblioteca:
    """Classe base (pai)"""
    def __init__(self, titulo, autor, ano):
        self.titulo = titulo
        self.autor = autor
        self.ano = ano
        self.emprestado = False

    def emprestar(self):
        self.emprestado = True

    def devolver(self):
        self.emprestado = False

    def info(self):
        return f"'{self.titulo}' por {self.autor} ({self.ano})"


class LivroFisico(ItemBiblioteca):
    """Herda de ItemBiblioteca"""
    def __init__(self, titulo, autor, ano, paginas, isbn):
        # Chama o __init__ do pai
        super().__init__(titulo, autor, ano)
        self.paginas = paginas
        self.isbn = isbn

    def info(self):
        # Estende o método do pai
        base = super().info()
        return f"{base} — {self.paginas} págs., ISBN: {self.isbn}"


class Ebook(ItemBiblioteca):
    def __init__(self, titulo, autor, ano, formato, tamanho_mb):
        super().__init__(titulo, autor, ano)
        self.formato = formato        # PDF, EPUB, MOBI
        self.tamanho_mb = tamanho_mb

    def info(self):
        base = super().info()
        return f"{base} — {self.formato}, {self.tamanho_mb}MB"


# Usando
fisico = LivroFisico("Clean Code", "Robert C. Martin", 2008, 464, "978-0132350884")
ebook = Ebook("Python Fluente", "Luciano Ramalho", 2015, "PDF", 6.2)

print(fisico.info())
# 'Clean Code' por Robert C. Martin (2008) — 464 págs., ISBN: 978-0132350884
print(ebook.info())
# 'Python Fluente' por Luciano Ramalho (2015) — PDF, 6.2MB

fisico.emprestar()   # Herdou de ItemBiblioteca
print(fisico.emprestado)  # True

5.2 — Herança Múltipla e MRO

Python permite herdar de várias classes ao mesmo tempo. O MRO (Method Resolution Order) define a ordem em que o Python busca métodos quando há conflito de nomes. 


class Emprestavel:
    """Mixin: adiciona funcionalidade de empréstimo"""
    def emprestar(self, usuario):
        self.emprestado_por = usuario
        print(f"Emprestado para {usuario}")

    def devolver(self):
        self.emprestado_por = None
        print("Devolvido")


class Avaliavel:
    """Mixin: adiciona sistema de avaliação"""
    def __init__(self):
        self._notas = []

    def avaliar(self, nota):
        if 1 <= nota <= 5:
            self._notas.append(nota)

    @property
    def media(self):
        if not self._notas:
            return 0
        return sum(self._notas) / len(self._notas)


class LivroCompleto(ItemBiblioteca, Emprestavel, Avaliavel):
    """Herança múltipla: combina 3 classes"""
    def __init__(self, titulo, autor, ano, paginas):
        ItemBiblioteca.__init__(self, titulo, autor, ano)
        Avaliavel.__init__(self)   # Inicializa o mixin de avaliação
        self.paginas = paginas


# Vendo a ordem de resolução
print(LivroCompleto.__mro__)
# LivroCompleto → ItemBiblioteca → Emprestavel → Avaliavel → object

livro = LivroCompleto("Design Patterns", "GoF", 1994, 416)
livro.emprestar("Maria")          # de Emprestavel
livro.avaliar(5)                  # de Avaliavel
livro.avaliar(4)
print(f"Média: {livro.media}")   # 4.5
print(livro.info())              # de ItemBiblioteca

Regra do MRO: Python segue a ordem da esquerda para direita na declaração da classe, usando o algoritmo C3 Linearization. Classes mais à esquerda têm prioridade.

6. Polimorfismo

Polimorfismo significa "muitas formas". O mesmo nome de método pode ter comportamentos diferentes em classes diferentes. 


class Livro:
    def calcular_multa(self, dias_atraso):
        return dias_atraso * 2.00   # R$ 2,00 por dia

class Revista:
    def calcular_multa(self, dias_atraso):
        return dias_atraso * 1.00   # R$ 1,00 por dia (mais barato)

class DVD:
    def calcular_multa(self, dias_atraso):
        return dias_atraso * 5.00   # R$ 5,00 por dia (mais caro)


# Polimorfismo: a função não sabe (nem precisa saber) o tipo
def cobrar_multa(item, dias):
    valor = item.calcular_multa(dias)
    print(f"Multa por {dias} dias de atraso: R$ {valor:.2f}")
    return valor


itens = [Livro(), Revista(), DVD()]
for item in itens:
    cobrar_multa(item, 3)

# Multa por 3 dias de atraso: R$ 6.00   (Livro)
# Multa por 3 dias de atraso: R$ 3.00   (Revista)
# Multa por 3 dias de atraso: R$ 15.00  (DVD)

7. Sobrecarga de Operadores

Você pode definir como os operadores (+, -, >, etc.) se comportam com seus objetos. 


class Livro:
    def __init__(self, titulo, paginas, preco):
        self.titulo = titulo
        self.paginas = paginas
        self.preco = preco

    def __str__(self):
        return f"'{self.titulo}' ({self.paginas} págs, R$ {self.preco:.2f})"

    def __add__(self, outro):
        """livro1 + livro2 → soma de páginas"""
        if isinstance(outro, Livro):
            return self.paginas + outro.paginas
        return NotImplemented

    def __gt__(self, outro):
        """livro1 > livro2 → compara por páginas"""
        if isinstance(outro, Livro):
            return self.paginas > outro.paginas
        return NotImplemented

    def __eq__(self, outro):
        """livro1 == livro2 → mesmo título e autor?"""
        if isinstance(outro, Livro):
            return self.titulo == outro.titulo
        return False


livro1 = Livro("Dom Casmurro", 256, 39.90)
livro2 = Livro("1984", 328, 45.00)

print(livro1 + livro2)    # 584 (soma de páginas)
print(livro1 > livro2)    # False (256 não é > 328)
print(livro1 == livro2)   # False (títulos diferentes)

livro3 = Livro("Dom Casmurro", 200, 29.90)
print(livro1 == livro3)   # True (mesmo título)

Principais operadores que você pode sobrecarregar

Operador Método Exemplo
+ __add__ a + b
- __sub__ a - b
* __mul__ a * b
== __eq__ a == b
!= __ne__ a != b
< __lt__ a < b
> __gt__ a > b
len() __len__ len(a)
in __contains__ x in a
[] __getitem__ a[0]

from datetime import datetime, timedelta

# CLASSE BASE E SUBCLASSES

class ItemBiblioteca:
    """Classe base para todos os itens do acervo"""
    def __init__(self, titulo, autor, ano):
        self.titulo = titulo
        self.autor = autor
        self.ano = ano
        self._emprestado = False

    @property
    def emprestado(self):
        return self._emprestado

    def emprestar(self):
        if self._emprestado:
            return False
        self._emprestado = True
        return True

    def devolver(self):
        self._emprestado = False

    def calcular_multa(self, dias_atraso):
        raise NotImplementedError("Subclasses devem implementar")

    def __str__(self):
        status = "📕 Emprestado" if self._emprestado else "📗 Disponível"
        return f"{status} | '{self.titulo}' por {self.autor} ({self.ano})"

    def __eq__(self, outro):
        if isinstance(outro, ItemBiblioteca):
            return self.titulo == outro.titulo and self.autor == outro.autor
        return False


class Livro(ItemBiblioteca):
    def __init__(self, titulo, autor, ano, paginas, isbn):
        super().__init__(titulo, autor, ano)
        self.paginas = paginas
        self.isbn = isbn

    def calcular_multa(self, dias_atraso):
        return dias_atraso * 2.00

    def __str__(self):
        base = super().__str__()
        return f"{base} | {self.paginas} págs | ISBN: {self.isbn}"


class Revista(ItemBiblioteca):
    def __init__(self, titulo, autor, ano, edicao):
        super().__init__(titulo, autor, ano)
        self.edicao = edicao

    def calcular_multa(self, dias_atraso):
        return dias_atraso * 1.00

    def __str__(self):
        base = super().__str__()
        return f"{base} | Edição {self.edicao}"


class DVD(ItemBiblioteca):
    def __init__(self, titulo, autor, ano, duracao_min):
        super().__init__(titulo, autor, ano)
        self.duracao_min = duracao_min

    def calcular_multa(self, dias_atraso):
        return dias_atraso * 5.00

    def __str__(self):
        base = super().__str__()

class Usuario:
    def __init__(self, nome, email, matricula):
        self.nome = nome
        self._email = None
        self.email = email
        self.matricula = matricula
        self.emprestimos = []   # lista de (ItemBiblioteca, data_emprestimo)

    @property
    def email(self):
        return self._email

    @email.setter
    def email(self, valor):
        if '@' not in valor:
            raise ValueError("Email inválido")
        self._email = valor.lower()

    @property
    def itens_emprestados(self):
        return len(self.emprestimos)

    def __str__(self):
        return f"👤 {self.nome} ({self.matricula}) — {self.itens_emprestados} itens"

    def __eq__(self, outro):
        if isinstance(outro, Usuario):
            return self.matricula == outro.matricula
        return False


# ============================================================
# BIBLIOTECA (GERENCIADOR PRINCIPAL)
# ============================================================
class Biblioteca:
    def __init__(self, nome):
        self.nome = nome
        self._acervo = []       # lista de ItemBiblioteca
        self._usuarios = []     # lista de Usuario
        self._emprestimos = []  # lista de (usuario, item, data)

    # --- Métodos de acervo ---
    def adicionar_item(self, item):
        self._acervo.append(item)
        print(f"✅ Adicionado: {item.titulo}")

    def buscar_por_titulo(self, termo):
        resultados = [item for item in self._acervo
                      if termo.lower() in item.titulo.lower()]
        return resultados

    def listar_disponiveis(self):
        return [item for item in self._acervo if not item.emprestado]

    def listar_todos(self):
        return self._acervo

    # --- Métodos de usuário ---
    def cadastrar_usuario(self, usuario):
        if usuario in self._usuarios:
            print(f"❌ Usuário {usuario.matricula} já cadastrado.")
            return False
        self._usuarios.append(usuario)
        print(f"✅ Usuário cadastrado: {usuario.nome}")
        return True

    def buscar_usuario(self, matricula):
        for usuario in self._usuarios:
            if usuario.matricula == matricula:
                return usuario
        return None

    # --- Métodos de empréstimo ---
    def emprestar(self, matricula, titulo):
        usuario = self.buscar_usuario(matricula)
        if not usuario:
            print(f"❌ Usuário {matricula} não encontrado.")
            return False

        # Busca o item pelo título
        for item in self._acervo:
            if item.titulo.lower() == titulo.lower():
                if item.emprestado:
                    print(f"❌ '{item.titulo}' já está emprestado.")
                    return False

                if item.emprestar():
                    usuario.emprestimos.append((item, datetime.now()))
                    self._emprestimos.append((usuario, item, datetime.now()))
                    print(f"✅ '{item.titulo}' emprestado para {usuario.nome}.")
                    return True

        print(f"❌ Item '{titulo}' não encontrado no acervo.")
        return False

    def devolver(self, matricula, titulo):
        usuario = self.buscar_usuario(matricula)
        if not usuario:
            print(f"❌ Usuário {matricula} não encontrado.")
            return False

        for emp in usuario.emprestimos:
            item, data_emprestimo = emp
            if item.titulo.lower() == titulo.lower():
                dias = (datetime.now() - data_emprestimo).days
                if dias > 14:   # prazo de 14 dias
                    multa = item.calcular_multa(dias - 14)
                    print(f"⚠️  Atraso de {dias - 14} dias. Multa: R$ {multa:.2f}")
                else:
                    print(f"✅ Devolvido no prazo ({dias} dias).")

                item.devolver()
                usuario.emprestimos.remove(emp)
                return True

        print(f"❌ Usuário não possui '{titulo}' emprestado.")
        return False

    # --- Relatórios ---
    def relatorio_acervo(self):
        print(f"\n📋 Acervo da {self.nome}:")
        print(f"   Total: {len(self._acervo)} itens")
        disponiveis = len(self.listar_disponiveis())
        print(f"   Disponíveis: {disponiveis}")
        print(f"   Emprestados: {len(self._acervo) - disponiveis}")
        print()
        for item in self._acervo:
            print(f"   {item}")

    # --- Métodos mágicos ---
    def __len__(self):
        return len(self._acervo)

    def __contains__(self, titulo):
        return any(item.titulo.lower() == titulo.lower() for item in self._acervo)

    def __str__(self):
        return f"📚 {self.nome} — {len(self)} itens no acervo"

            else:

        elif opcao == "3":
            termo = input("Digite o título ou parte dele: ").strip()
            resultados = bib.buscar_por_titulo(termo)
            if resultados:
                print(f"\n🔍 {len(resultados)} resultado(s):")
                for item in resultados:
                    print(f"   {item}")
            else:
                print(f"❌ Nenhum item encontrado para '{termo}'.")

        elif opcao == "4":
            matricula = input("Matrícula do usuário: ").strip()
            titulo = input("Título do item: ").strip()

            titulo = input("Título do item: ").strip()

            email = input("Email: ").strip()
            matricula = input("Matrícula: ").strip()
            try:
                bib.cadastrar_usuario(Usuario(nome, email, matricula))
            except ValueError as e:
                print(f"❌ Erro: {e}")

        elif opcao == "7":
            bib.relatorio_acervo()

        elif opcao == "0":
            print("👋 Até logo!")
            break

        else:
            print("❌ Opção inválida!")

2. Adicione histórico de empréstimos no Usuario

Modifique a classe Usuario para manter um histórico completo de empréstimos (incluindo os já devolvidos). Adicione o método exibir_historico() que mostra todos os empréstimos do usuário, com data e status.

Gabarito: 

class Usuario:
    def __init__(self, nome, email, matricula):
        self.nome = nome
        self._email = None
        self.email = email
        self.matricula = matricula
        self.emprestimos = []
        self._historico = []   # (item, data_emprestimo, data_devolucao)

    def registrar_historico(self, item, data_emp, data_dev=None):
        self._historico.append({
            'item': item,
            'emprestimo': data_emp,
            'devolucao': data_dev
        })

    def exibir_historico(self):
        print(f"\n📜 Histórico de {self.nome}:")
        if not self._historico:
            print("   Nenhum empréstimo registrado.")
            return

        for registro in self._historico:
            item = registro['item']
            data_emp = registro['emprestimo'].strftime("%d/%m/%Y")
            if registro['devolucao']:
                data_dev = registro['devolucao'].strftime("%d/%m/%Y")
                print(f"   📗 {item.titulo} — {data_emp} → {data_dev} (devolvido)")
            else:
                print(f"   📕 {item.titulo} — {data_emp} → em andamento")
